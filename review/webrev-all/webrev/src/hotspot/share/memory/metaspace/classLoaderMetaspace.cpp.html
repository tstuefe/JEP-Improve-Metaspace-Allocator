  55 static bool use_class_space(bool is_class) {
  56   if (Metaspace::using_class_space() && is_class) {
  57     return true;
  58   }
  59   return false;
  60 }
  61 
  62 static bool use_class_space(Metaspace::MetadataType mdType) {
  63   return use_class_space(metaspace::is_class(mdType));
  64 }
  65 
  66 ClassLoaderMetaspace::ClassLoaderMetaspace(Mutex* lock, Metaspace::MetaspaceType space_type)
  67   : _lock(lock)
  68   , _space_type(space_type)
  69   , _non_class_space_arena(NULL)
  70   , _class_space_arena(NULL)
  71 {
  72   ChunkManager* const non_class_cm =
  73           ChunkManager::chunkmanager_nonclass();
  74 
  75   // Initialize non-class Arena
  76   _non_class_space_arena = new MetaspaceArena(
  77       non_class_cm,
  78       ArenaGrowthPolicy::policy_for_space_type(space_type, false),
  79       lock,
  80       RunningCounters::used_nonclass_counter(),
  81       "non-class sm");
  82 
  83   // If needed, initialize class arena
  84   if (Metaspace::using_class_space()) {
  85     ChunkManager* const class_cm =
  86             ChunkManager::chunkmanager_class();
  87     _class_space_arena = new MetaspaceArena(
  88         class_cm,
  89         ArenaGrowthPolicy::policy_for_space_type(space_type, true),
  90         lock,
  91         RunningCounters::used_class_counter(),
  92         "class sm");
  93   }
  94 
  95   UL2(debug, "born (SpcMgr nonclass: " PTR_FORMAT ", SpcMgr class: " PTR_FORMAT ".",
  96       p2i(_non_class_space_arena), p2i(_class_space_arena));
  97 }
  98 
  99 ClassLoaderMetaspace::~ClassLoaderMetaspace() {
 100   Metaspace::assert_not_frozen();
 101 
 102   UL(debug, "dies.");
 103 
 104   delete _non_class_space_arena;
 105   delete _class_space_arena;
 106 
 107 }

 109 // Allocate word_size words from Metaspace.
 110 MetaWord* ClassLoaderMetaspace::allocate(size_t word_size, Metaspace::MetadataType mdType) {
 111   Metaspace::assert_not_frozen();
 112   if (use_class_space(mdType)) {
 113     return class_space_arena()->allocate(word_size);
 114   } else {
 115     return non_class_space_arena()->allocate(word_size);
 116   }
 117 }
 118 
 119 // Attempt to expand the GC threshold to be good for at least another word_size words
 120 // and allocate. Returns NULL if failure. Used during Metaspace GC.
 121 MetaWord* ClassLoaderMetaspace::expand_and_allocate(size_t word_size, Metaspace::MetadataType mdType) {
 122   Metaspace::assert_not_frozen();
 123   size_t delta_bytes = MetaspaceGC::delta_capacity_until_GC(word_size * BytesPerWord);
 124   assert(delta_bytes > 0, "Must be");
 125 
 126   size_t before = 0;
 127   size_t after = 0;
 128   bool can_retry = true;
 129   MetaWord* res;
 130   bool incremented;
 131 
 132   // Each thread increments the HWM at most once. Even if the thread fails to increment
 133   // the HWM, an allocation is still attempted. This is because another thread must then
 134   // have incremented the HWM and therefore the allocation might still succeed.
 135   do {
 136     incremented = MetaspaceGC::inc_capacity_until_GC(delta_bytes, &after, &before, &can_retry);
 137     res = allocate(word_size, mdType);
 138   } while (!incremented && res == NULL && can_retry);
 139 
 140   if (incremented) {
 141     Metaspace::tracer()->report_gc_threshold(before, after,
 142                                   MetaspaceGCThresholdUpdater::ExpandAndAllocate);
 143     // Keeping both for now until I am sure the old variant (gc + metaspace) is not needed anymore
 144     log_trace(gc, metaspace)("Increase capacity to GC from " SIZE_FORMAT " to " SIZE_FORMAT, before, after);
 145     UL2(info, "GC threshold increased: " SIZE_FORMAT "->" SIZE_FORMAT ".", before, after);
 146   }
 147 
 148   return res;
 149 }

 151 // Prematurely returns a metaspace allocation to the _block_freelists
 152 // because it is not needed anymore.
 153 void ClassLoaderMetaspace::deallocate(MetaWord* ptr, size_t word_size, bool is_class) {
 154 
 155   Metaspace::assert_not_frozen();
 156 
 157   if (use_class_space(is_class)) {
 158     class_space_arena()->deallocate(ptr, word_size);
 159   } else {
 160     non_class_space_arena()->deallocate(ptr, word_size);
 161   }
 162 
 163   DEBUG_ONLY(InternalStats::inc_num_deallocs();)
 164 
 165 }

 167 // Update statistics. This walks all in-use chunks.
 168 void ClassLoaderMetaspace::add_to_statistics(clms_stats_t* out) const {
 169   if (non_class_space_arena() != NULL) {
 170     non_class_space_arena()->add_to_statistics(&out->arena_stats_nonclass);
 171   }
 172   if (class_space_arena() != NULL) {
 173     class_space_arena()->add_to_statistics(&out->arena_stats_class);
 174   }
 175 }

 177 #ifdef ASSERT
 178 void ClassLoaderMetaspace::verify() const {
 179   metaspace::check_valid_spacetype(_space_type);
 180   if (non_class_space_arena() != NULL) {
 181     non_class_space_arena()->verify(false);
 182   }
 183   if (class_space_arena() != NULL) {
 184     class_space_arena()->verify(false);
 185   }
 186 }

 190 // This only exists for JFR and jcmd VM.classloader_stats. We may want to
 191 //  change this. Capacity as a stat is of questionable use since it may
 192 //  contain committed and uncommitted areas. For now we do this to maintain
 193 //  backward compatibility with JFR.
 194 void ClassLoaderMetaspace::calculate_jfr_stats(size_t* p_used_bytes, size_t* p_capacity_bytes) const {
 195   // Implement this using the standard statistics objects.
 196   size_t used_c = 0, cap_c = 0, used_nc = 0, cap_nc = 0;
 197   if (non_class_space_arena() != NULL) {
 198     non_class_space_arena()->usage_numbers(&used_nc, NULL, &cap_nc);
 199   }
 200   if (class_space_arena() != NULL) {
 201     class_space_arena()->usage_numbers(&used_c, NULL, &cap_c);
 202   }
 203   if (p_used_bytes != NULL) {
 204     *p_used_bytes = used_c + used_nc;
 205   }
 206   if (p_capacity_bytes != NULL) {
 207     *p_capacity_bytes = cap_c + cap_nc;
 208   }
 209 }
