  25 #ifndef SHARE_MEMORY_METASPACE_METABLOCK_HPP
  26 #define SHARE_MEMORY_METASPACE_METABLOCK_HPP
  27 
  28 #include "memory/metaspace/metabase.hpp"
  29 #include "utilities/globalDefinitions.hpp"
  30 
  31 namespace metaspace {
  32 
  33 // Metablock is the unit of allocation from a Chunk.
  34 //
  35 // A Metablock may be reused by its SpaceManager but are never moved between
  36 // SpaceManagers.  There is no explicit link to the Metachunk
  37 // from which it was allocated.  Metablock may be deallocated and
  38 // put on a freelist but the space is never freed, rather
  39 // the Metachunk it is a part of will be deallocated when it's
  40 // associated class loader is collected.
  41 
  42 class Metablock : public Metabase<Metablock> {
  43   friend class VMStructs;
  44  public:
  45   Metablock(size_t word_size) : Metabase<Metablock>(word_size) {}
  46 };
  47 
  48 } // namespace metaspace
  49 
  50 #endif // SHARE_MEMORY_METASPACE_METABLOCK_HPP
