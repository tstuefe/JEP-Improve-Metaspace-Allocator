3086 static int recursiveFindType(VMTypeEntry* origtypes, const char* typeName, bool isRecurse) {
3087   {
3088     VMTypeEntry* types = origtypes;
3089     while (types->typeName != NULL) {
3090       if (strcmp(typeName, types->typeName) == 0) {
3091         // Found it
3092         return 1;
3093       }
3094       ++types;
3095     }
3096   }
3097   // Search for the base type by peeling off const and *
3098   size_t len = strlen(typeName);
3099   if (typeName[len-1] == '*') {
3100     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3101     strncpy(s, typeName, len - 1);
3102     s[len-1] = '\0';
3103     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3104     if (recursiveFindType(origtypes, s, true) == 1) {
3105       FREE_C_HEAP_ARRAY(char, s);
3106       return 1;
3107     }
3108     FREE_C_HEAP_ARRAY(char, s);
3109   }
3110   const char* start = NULL;
3111   if (strstr(typeName, "GrowableArray<") == typeName) {
3112     start = typeName + strlen("GrowableArray<");
3113   } else if (strstr(typeName, "Array<") == typeName) {
3114     start = typeName + strlen("Array<");
3115   }
3116   if (start != NULL) {
3117     const char * end = strrchr(typeName, '>');
3118     int len = end - start + 1;
3119     char * s = NEW_C_HEAP_ARRAY(char, len, mtInternal);
3120     strncpy(s, start, len - 1);
3121     s[len-1] = '\0';
3122     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3123     if (recursiveFindType(origtypes, s, true) == 1) {
3124       FREE_C_HEAP_ARRAY(char, s);
3125       return 1;
3126     }
3127     FREE_C_HEAP_ARRAY(char, s);
3128   }
3129   if (strstr(typeName, "const ") == typeName) {
3130     const char * s = typeName + strlen("const ");
3131     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3132     if (recursiveFindType(origtypes, s, true) == 1) {
3133       return 1;
3134     }
3135   }
3136   if (strstr(typeName, " const") == typeName + len - 6) {
3137     char * s = os::strdup_check_oom(typeName);
3138     s[len - 6] = '\0';
3139     // tty->print_cr("checking \"%s\" for \"%s\"", s, typeName);
3140     if (recursiveFindType(origtypes, s, true) == 1) {
3141       os::free(s);
3142       return 1;
3143     }
3144     os::free(s);
3145   }
3146   if (!isRecurse) {
3147     tty->print_cr("type \"%s\" not found", typeName);
3148   }
3149   return 0;
3150 }
