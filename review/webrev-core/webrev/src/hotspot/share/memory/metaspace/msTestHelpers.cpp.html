  60 MetaspaceTestContext::MetaspaceTestContext(const char* name, size_t commit_limit, size_t reserve_limit)
  61   : _name(name), _reserve_limit(reserve_limit), _commit_limit(commit_limit),
  62     _context(NULL),
  63     _commit_limiter(commit_limit == 0 ? max_uintx : commit_limit), // commit_limit == 0 -> no limit
  64     _used_words_counter()
  65 {
  66   assert(is_aligned(reserve_limit, Metaspace::reserve_alignment_words()), "reserve_limit (" SIZE_FORMAT ") "
  67                     "not aligned to metaspace reserve alignment (" SIZE_FORMAT ")",
  68                     reserve_limit, Metaspace::reserve_alignment_words());
  69   if (reserve_limit > 0) {
  70     // have reserve limit -> non-expandable context
  71     ReservedSpace rs(reserve_limit * BytesPerWord, Metaspace::reserve_alignment(), false);
  72     _context = MetaspaceContext::create_nonexpandable_context(name, rs, &_commit_limiter);
  73   } else {
  74     // no reserve limit -> expandable vslist
  75     _context = MetaspaceContext::create_expandable_context(name, &_commit_limiter);
  76   }
  77 
  78 }
  79 
  80 MetaspaceTestContext::~MetaspaceTestContext() {
  81   DEBUG_ONLY(verify();)
  82   MutexLocker fcl(MetaspaceExpand_lock, Mutex::_no_safepoint_check_flag);
  83   delete _context;
  84 }
  85 
  86 // Create an arena, feeding off this area.
  87 MetaspaceTestArena* MetaspaceTestContext::create_arena(Metaspace::MetaspaceType type) {
  88   const ArenaGrowthPolicy* growth_policy = ArenaGrowthPolicy::policy_for_space_type(type, false);
  89   Mutex* lock = new Mutex(Monitor::native, "MetaspaceTestArea-lock", false, Monitor::_safepoint_check_never);
  90   MetaspaceArena* arena = NULL;
  91   {
  92     MutexLocker ml(lock,  Mutex::_no_safepoint_check_flag);
  93     arena = new MetaspaceArena(_context->cm(), growth_policy, lock, &_used_words_counter, _name);
  94   }
  95   return new MetaspaceTestArena(lock, arena);
  96 }
  97 
  98 void MetaspaceTestContext::purge_area() {
  99   _context->cm()->purge();
 100 }
