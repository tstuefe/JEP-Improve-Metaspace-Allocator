  26 #ifndef SHARE_MEMORY_METASPACE_MSREPORT_HPP
  27 #define SHARE_MEMORY_METASPACE_MSREPORT_HPP
  28 
  29 #include "memory/allocation.hpp"
  30 
  31 namespace metaspace {
  32 
  33 class MetaspaceReporter : public AllStatic {
  34 public:
  35 
  36   // Flags for print_report().
  37   enum class Option {
  38     // Show usage by class loader.
  39     ShowLoaders                 = (1 << 0),
  40     // Breaks report down by chunk type (small, medium, ...).
  41     BreakDownByChunkType        = (1 << 1),
  42     // Breaks report down by space type (anonymous, reflection, ...).
  43     BreakDownBySpaceType        = (1 << 2),
  44     // Print details about the underlying virtual spaces.
  45     ShowVSList                  = (1 << 3),
  46     // If show_loaders: show loaded classes for each loader.
  47     ShowClasses                 = (1 << 4)
  48   };
  49 
  50   // This will print out a basic metaspace usage report but
  51   // unlike print_report() is guaranteed not to lock or to walk the CLDG.
  52   static void print_basic_report(outputStream* st, size_t scale);
  53 
  54   // Prints a report about the current metaspace state.
  55   // Optional parts can be enabled via flags.
  56   // Function will walk the CLDG and will lock the expand lock; if that is not
  57   // convenient, use print_basic_report() instead.
  58   static void print_report(outputStream* out, size_t scale = 0, int flags = 0);
  59 
  60 };
  61 
  62 } // namespace metaspace
  63 
  64 #endif // SHARE_MEMORY_METASPACE_MSREPORT_HPP
