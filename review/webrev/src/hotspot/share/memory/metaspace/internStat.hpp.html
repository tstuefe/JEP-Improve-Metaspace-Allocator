  25 #ifndef SHARE_MEMORY_METASPACE_INTERNSTAT_HPP
  26 #define SHARE_MEMORY_METASPACE_INTERNSTAT_HPP
  27 
  28 #include "memory/allocation.hpp"
  29 #include "runtime/atomic.hpp"
  30 #include "utilities/globalDefinitions.hpp"
  31 
  32 class outputStream;
  33 
  34 namespace metaspace {
  35 
  36 
  37 // These are some counters useful for debugging and analyzing Metaspace problems.
  38 // They get printed as part of the Metaspace report (e.g. via jcmd VM.metaspace)
  39 
  40 class InternalStats : public AllStatic {
  41 
  42   // Note: all counters which are modified on the classloader local allocation path
  43   //   (not under ExpandLock protection) have to be atomic.
  44 
  45 #define ALL_MY_COUNTERS(x, x_atomic)                \
  46                                                     \
  47   /* Number of allocations. */                      \
  48   DEBUG_ONLY(x_atomic(num_allocs))                  \
  49                                                     \
  50   /* Number of external deallocations */            \
  51   /* (excluding retired chunk remains) */           \
  52         DEBUG_ONLY(x_atomic(num_deallocs))                \
  53                                                     \
  54   /* Number of times an allocation was satisfied */ \
  55   /*  from deallocated blocks. */                   \
  56   DEBUG_ONLY(x_atomic(num_allocs_from_deallocated_blocks)) \
  57                                                     \
  58   /* Number of times an arena retired a chunk */    \
  59   DEBUG_ONLY(x_atomic(num_chunks_retired))          \
  60                                                     \
  61   /* Number of times an allocation failed */        \
  62   /*  because we hit a limit. */                    \
  63   x_atomic(num_allocs_failed_limit)                 \
  64                                                     \
  65   /* Number of times an arena was born ... */       \
  66         x_atomic(num_arena_births)                        \
  67   /* ... and died. */                               \
  68         x_atomic(num_arena_deaths)                        \
  69                                                     \
  70         /* Number of times a micro arena (anon/hidden */  \
  71         /*  class or reflection proxy) was born. */       \
  72   DEBUG_ONLY(x_atomic(num_micro_arena_births))      \
  73         /* ... and died. */                               \
  74         DEBUG_ONLY(x_atomic(num_micro_arena_deaths))      \
  75                                                     \
  76   /* Number of times VirtualSpaceNode were */       \
  77   /*  born...  */                                   \
  78   x(num_vsnodes_births)                             \
  79   /* ... and died. */                               \
  80   x(num_vsnodes_deaths)                             \
  81                                                     \
  82   /* Number of times we committed space. */         \
  83   x(num_space_committed)                            \
  84   /* Number of times we uncommitted space. */       \
  85   x(num_space_uncommitted)                          \
  86                                                     \
  87   /* Number of times a chunk was returned to the */ \
  88   /*  freelist (external only). */                  \
  89   x(num_chunks_returned_to_freelist)                \
  90   /* Number of times a chunk was taken from */      \
  91   /*  freelist (external only) */                   \
  92   x(num_chunks_taken_from_freelist)                 \
  93                                                     \
  94   /* Number of successful chunk merges */           \
  95   x(num_chunk_merges)                               \
  96   /* Number of chunk splits */                      \
  97   x(num_chunk_splits)                               \
  98   /* Number of chunk in place enlargements */       \
  99   x(num_chunks_enlarged)                            \
 100                                                     \
 101   /* Number of times we did a purge */              \
 102   x(num_purges)                                     \
 103 
 104 
 105 #define DEFINE_COUNTER(name)          static uint64_t _##name;
 106 #define DEFINE_ATOMIC_COUNTER(name)   static volatile uint64_t _##name;
 107   ALL_MY_COUNTERS(DEFINE_COUNTER, DEFINE_ATOMIC_COUNTER)
 108 #undef DEFINE_COUNTER
 109 #undef DEFINE_ATOMIC_COUNTER
 110 
 111 public:
 112 
 113 // incrementors
 114 #define INCREMENTOR(name)           static void inc_##name() { _##name ++; }
 115 #define INCREMENTOR_ATOMIC(name)    static void inc_##name() { Atomic::inc(&_##name); }
 116   ALL_MY_COUNTERS(INCREMENTOR, INCREMENTOR_ATOMIC)
 117 #undef INCREMENTOR
 118 #undef INCREMENTOR_ATOMIC
 119 
 120 // getters
 121 #define GETTER(name)                static uint64_t name() { return _##name; }
 122   ALL_MY_COUNTERS(GETTER, GETTER)
 123 #undef GETTER
 124 
 125 
 126   static void print_on(outputStream* st);
 127 
 128 
 129 };
 130 
 131 } // namespace metaspace
 132 
 133 #endif // SHARE_MEMORY_METASPACE_INTERNSTAT_HPP
