  43 // Class FreeBlocks manages deallocated blocks in Metaspace.
  44 //
  45 // In Metaspace, deallocations are an uncommon occurrence of
  46 // allocated memory blocks which are deallocated prematurely. Normally,
  47 // memory blocks stay allocated as long as the ClassLoaderMetaspace itself
  48 // exists - i.e. typically, until the class loader is unloaded.
  49 //
  50 // However, there are cases when Metaspace memory blocks are deallocated
  51 // prematurely. E.g. when class loading errors happen and half-loaded
  52 // metadata are left over; or when a class is redefined, leaving the
  53 // old bytecode. For details, see Metaspace::deallocate().
  54 //
  55 // All these blocks can be reused, so they are collected. Since these
  56 // blocks are embedded into chunks which are still in use by a very alive
  57 // class loader, we cannot give these blocks to other class loaders; we can
  58 // however collect these blocks at the class loader level and reuse them
  59 // for future allocations from the same class loader.
  60 //
  61 // This is what FreeBlocks does: it manages small memory blocks.
  62 //
  63 // FreeBlocks is optimized toward the typical size and number of deallocated
  64 //  blocks. The vast majority of them (about 90%) are below 16 words in size,
  65 //  but there is a significant portion of memory blocks much larger than that -
  66 //  leftover space from used uu chunks, see MetaspaceArena::retire_current_chunk().
  67 //
  68 
  69 class FreeBlocks : public CHeapObj<mtMetaspace> {
  70 
  71   typedef BinList32 SmallBlocksType;
  72 
  73   // _small_blocks takes care of small to very small blocks.
  74   SmallBlocksType _small_blocks;
  75 
  76   // A BST for larger blocks.
  77   BlockTree _tree;
  78 
  79   static const size_t splinter_threshold = 0;// 0x100;
  80 
  81 public:
  82 
  83   const static size_t minimal_word_size = SmallBlocksType::minimal_word_size;
  84 
  85   // Add a block to the deallocation management.
  86   void add_block(MetaWord* p, size_t word_size);
  87 
  88   // Retrieve a block of at least requested_word_size.
  89   MetaWord* get_block(size_t requested_word_size);
  90 
  91 #ifdef ASSERT
  92   void verify() const {
  93     _tree.verify();
  94     _small_blocks.verify();
  95   };
  96 #endif
  97 
  98   // Returns number of blocks.
  99   int count() const {
 100     return _small_blocks.count() + _tree.count();
 101   }
 102 
 103   // Returns total size, in words, of all elements.
 104   size_t total_size() const {
 105     return _small_blocks.total_size() + _tree.total_size();
 106   }
 107 
 108   // Returns true if empty.
 109   bool is_empty() const {
 110     return _small_blocks.is_empty() && _tree.is_empty();
 111   }
 112 
 113 };
