  37 PrintMetaspaceInfoKlassClosure::PrintMetaspaceInfoKlassClosure(outputStream* out, bool do_print)
  38 : _out(out), _cnt(0)
  39 {}
  40 
  41 void PrintMetaspaceInfoKlassClosure::do_klass(Klass* k) {
  42   _cnt ++;
  43   _out->cr_indent();
  44   _out->print(UINTX_FORMAT_W(4) ": ", _cnt);
  45 
  46   // Print a 's' for shared classes
  47   _out->put(k->is_shared() ? 's': ' ');
  48 
  49   ResourceMark rm;
  50   _out->print("  %s", k->external_name());
  51 
  52   // Special treatment for generated core reflection accessor classes: print invocation target.
  53   if (ReflectionAccessorImplKlassHelper::is_generated_accessor(k)) {
  54     _out->print(" (invokes: ");
  55     ReflectionAccessorImplKlassHelper::print_invocation_target(_out, k);
  56     _out->print(")");
  57   }
  58 }
