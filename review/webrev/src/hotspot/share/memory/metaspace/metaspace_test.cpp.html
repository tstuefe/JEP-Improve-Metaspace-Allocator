  63 ///// MetaspaceTestArea //////

  65 MetaspaceTestContext::MetaspaceTestContext(const char* name, size_t commit_limit, size_t reserve_limit)
  66   : _name(name), _reserve_limit(reserve_limit), _commit_limit(commit_limit),
  67     _context(NULL),
  68     _commit_limiter(commit_limit == 0 ? max_uintx : commit_limit), // commit_limit == 0 -> no limit
  69     _used_words_counter()
  70 {
  71 
  72   if (reserve_limit > 0) {
  73     // have reserve limit -> non-expandable context
  74     ReservedSpace rs(reserve_limit * BytesPerWord, Metaspace::reserve_alignment(), false);
  75     _context = MetaspaceContext::create_nonexpandable_context(name, rs, &_commit_limiter);
  76   } else {
  77     // no reserve limit -> expandable vslist
  78     _context = MetaspaceContext::create_expandable_context(name, &_commit_limiter);
  79   }
  80 
  81 }
  82 
  83 MetaspaceTestContext::~MetaspaceTestContext() {
  84   DEBUG_ONLY(verify(true);)
  85   MutexLocker fcl(MetaspaceExpand_lock, Mutex::_no_safepoint_check_flag);
  86   delete _context;
  87 }
  88 
  89 // Create an arena, feeding off this area.
  90 MetaspaceTestArena* MetaspaceTestContext::create_arena(MetaspaceType type) {
  91   const ArenaGrowthPolicy* growth_policy = ArenaGrowthPolicy::policy_for_space_type(type, false);
  92   Mutex* lock = new Mutex(Monitor::native, "MetaspaceTestArea-lock", false, Monitor::_safepoint_check_never);
  93   MetaspaceArena* arena = NULL;
  94   {
  95     MutexLocker ml(lock,  Mutex::_no_safepoint_check_flag);
  96     arena = new MetaspaceArena(_context->cm(), growth_policy, lock, &_used_words_counter, _name, false);
  97   }
  98   return new MetaspaceTestArena(lock, arena);
  99 }

 102   _context->cm()->purge();
 103 }
