  39 bool MetachunkList::contains(const Metachunk* c) const {
  40   for (Metachunk* c2 = _first; c2 != NULL; c2 = c2->next()) {
  41     if (c == c2) {
  42       return true;
  43     }
  44   }
  45   return false;
  46 }
  47 
  48 void MetachunkList::verify() const {
  49   int num = 0;
  50   const Metachunk* last_c = NULL;
  51   for (const Metachunk* c = _first; c != NULL; c = c->next()) {
  52     num ++;
  53     assert(c->prev() != c && c->next() != c, "circularity");
  54     assert(c->prev() == last_c,
  55            "Broken link to predecessor. Chunk " METACHUNK_FULL_FORMAT ".",
  56            METACHUNK_FULL_FORMAT_ARGS(c));
  57     c->verify(false);
  58     last_c = c;
  59   }
  60   _num_chunks.check(num);
  61 }
  62 
  63 #endif // ASSERT

  66 size_t MetachunkList::calc_committed_word_size() const {
  67 
  68   if (_first != NULL && _first->is_dead()) {
  69     // list used for chunk header pool; dead chunks have no size.
  70     return 0;
  71   }
  72 
  73   size_t s = 0;
  74   for (Metachunk* c = _first; c != NULL; c = c->next()) {
  75     assert(c->is_dead() == false, "Sanity");
  76     s += c->committed_words();
  77   }
  78   return s;
  79 }
  80 
  81 size_t MetachunkList::calc_word_size() const {
  82 
  83   if (_first != NULL && _first->is_dead()) {
  84     // list used for chunk header pool; dead chunks have no size.
  85     return 0;
  86   }
  87 
  88   size_t s = 0;
  89   for (Metachunk* c = _first; c != NULL; c = c->next()) {
  90     assert(c->is_dead() == false, "Sanity");
  91     s += c->committed_words();
  92   }
  93   return s;
  94 
  95 }
  96 
  97 void MetachunkList::print_on(outputStream* st) const {
  98 
  99   if (_num_chunks.get() > 0) {
 100     for (const Metachunk* c = _first; c != NULL; c = c->next()) {
 101       st->print(" - <");
 102       c->print_on(st);
 103       st->print(">");
 104     }
 105     st->print(" - total : %d chunks.", _num_chunks.get());
 106   } else {
 107     st->print("empty");
 108   }
 109 
 110 }
