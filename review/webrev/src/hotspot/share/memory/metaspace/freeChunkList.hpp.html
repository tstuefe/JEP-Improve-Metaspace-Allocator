  40 // This is the free list underlying the ChunkManager.
  41 //
  42 // Chunks are kept in a vector of double-linked double-headed lists
  43 //  (using Metachunk::prev/next). One list per chunk level exists.
  44 //
  45 // Chunks in these lists are roughly ordered, in a fashion: uncommitted chunks
  46 //  are added to the back of the list, fully or partially committed chunks to the
  47 //  front. So, chunks at the front of the list are fully or partially committed, but
  48 //  within that group are unsorted.
  49 //
  50 // Note that there is a small caveat: while chunks can never get uncommitted when
  51 //  residing in the freelist, they *can* get committed as a side effect of committing
  52 //  neighboring chunks, if the chunk happens to reside in the same commit granule.
  53 //  In practice this is not a big deal and not enough motivation to reorder this
  54 //  list. For our purposes this list is "ordered enough".
  55 //
  56 // Beyond adding at either front or at back, we do not sort on insert, since the
  57 //  insert path is used during Metaspace reclamation which may happen at GC pause.
  58 //
  59 // During retrieval (at class loading), we search the list for a chunk of at least
  60 //  n committed words to satisfy the caller requested committed word size. We stop
  61 //  searching at the first fully uncommitted chunk.
  62 //
  63 // Note that even though this is an O(n) search, in practice this is not a problem:
  64 //  - in all likelihood the requested commit word size is way smaller than even a single
  65 //    commit granule, so 99% of all searches would end at the first chunk (which is either
  66 //    uncommitted or committed to at least one commit granule size).
  67 //  - in all likelihood chunks, when added to this list, are either fully committed
  68 //    or fully uncommitted (see Settings::uncommit_on_return_min_word_size()).
  69 //
  70 // Should we ever encounter situations where the O(n) search is a bottleneck, this
  71 //  structure can easily be optimized (e.g. a BST). But for now lets keep this simple.
  72 
  73 class FreeChunkList {
  74 
  75   Metachunk* _first;
  76   Metachunk* _last;
  77 
  78   IntCounter _num_chunks;
  79   SizeCounter _committed_word_size;
  80 
  81   void add_front(Metachunk* c) {
  82     if (_first == NULL) {
  83       assert(_last == NULL, "Sanity");
  84       _first = _last = c;
  85       c->set_prev(NULL);
  86       c->set_next(NULL);
  87     } else {
  88       assert(_last != NULL, "Sanity");
  89       c->set_next(_first);
  90       c->set_prev(NULL);
  91       _first->set_prev(c);
  92       _first = c;
  93     }
  94   }
  95 
  96   // Add chunk to the back of the list.
  97   void add_back(Metachunk* c) {
  98     if (_last == NULL) {
  99       assert(_first == NULL, "Sanity");
 100       _last = _first = c;
 101       c->set_prev(NULL);
 102       c->set_next(NULL);
 103     } else {
 104       assert(_first != NULL, "Sanity");
 105       c->set_next(NULL);
 106       c->set_prev(_last);
 107       _last->set_next(c);
 108       _last = c;
 109     }
 110   }
 111 
 112 public:
 113 
 114   FreeChunkList() :
 115     _first(NULL),
 116     _last(NULL)
 117     {}
 118 
 119   // Remove given chunk from anywhere in the list.
 120   Metachunk* remove(Metachunk* c) {
 121     assert(contains(c), "Must be contained here");
 122     Metachunk* pred = c->prev();
 123     Metachunk* succ = c->next();
 124     if (pred) {
 125       pred->set_next(succ);
 126     }
 127     if (succ) {
 128       succ->set_prev(pred);
 129     }
 130     if (_first == c) {
 131       _first = succ;
 132     }
 133     if (_last == c) {
 134       _last = pred;
 135     }
 136     c->set_next(NULL);
 137     c->set_prev(NULL);
 138     _committed_word_size.decrement_by(c->committed_words());
 139     _num_chunks.decrement();
 140     return c;
 141   }
 142 
 143   void add(Metachunk* c) {
 144     assert(contains(c) == false, "Chunk already in freelist");
 145     assert(_first == NULL || _first->level() == c->level(), "wrong level");
 146     // Uncomitted chunks go to the back, fully or partially committed to the front.
 147     if (c->committed_words() == 0) {
 148       add_back(c);
 149     } else {
 150       add_front(c);
 151     }
 152     _committed_word_size.increment_by(c->committed_words());
 153     _num_chunks.increment();
 154   }
 155 
 156   // Removes the first chunk from the list and returns it. Returns NULL if list is empty.
 157   Metachunk* remove_first() {
 158     Metachunk* c = _first;
 159     if (c != NULL) {
 160       remove(c);
 161     }
 162     return c;
 163   }
 164 
 165   // Find and removes a chunk in this list which has at least min_committed_words committed words.
 166   // Returns NULL if not found.
 167   Metachunk* find_matching(size_t min_committed_words) {
 168     Metachunk* c = _first;
 169     while (c != NULL && c->committed_words() > 0) {
 170       if (c->committed_words() <= min_committed_words) {
 171         remove(c);
 172         return c;
 173       }
 174       c = c->next();
 175     }
 176     return NULL;
 177   }
 178 
 179   // Returns reference to the first chunk in the list, or NULL
 180   Metachunk* first() const { return _first; }
 181 
 182 #ifdef ASSERT
 183   bool contains(const Metachunk* c) const;
 184   void verify() const;
 185 #endif
 186 
 187   // Returns number of chunks
 188   int num_chunks() const { return _num_chunks.get(); }
 189 
 190   // Returns total committed word size
 191   size_t committed_word_size() const { return _committed_word_size.get(); }
 192 
 193   void print_on(outputStream* st) const;
 194 
 195 };

 198 // A vector of free chunk lists, one per chunk level
 199 class FreeChunkListVector {
 200 
 201   FreeChunkList _lists[chunklevel::NUM_CHUNK_LEVELS];
 202 
 203   const FreeChunkList* list_for_level(chunklevel_t lvl) const         { DEBUG_ONLY(chunklevel::check_valid_level(lvl)); return _lists + lvl; }
 204   FreeChunkList* list_for_level(chunklevel_t lvl)                     { DEBUG_ONLY(chunklevel::check_valid_level(lvl)); return _lists + lvl; }
 205 
 206   const FreeChunkList* list_for_chunk(const Metachunk* c) const       { return list_for_level(c->level()); }
 207   FreeChunkList* list_for_chunk(const Metachunk* c)                   { return list_for_level(c->level()); }
 208 
 209 public:
 210 
 211   // Remove given chunk from its list. List must contain that chunk.
 212   void remove(Metachunk* c) {
 213     list_for_chunk(c)->remove(c);
 214   }
 215 
 216   // Remove first node unless empty. Returns node or NULL.
 217   Metachunk* remove_first(chunklevel_t lvl) {
 218     Metachunk* c = list_for_level(lvl)->remove_first();
 219     return c;
 220   }
 221 
 222   void add(Metachunk* c) {
 223     list_for_chunk(c)->add(c);
 224   }
 225 
 226   // Returns number of chunks for a given level.
 227   int num_chunks_at_level(chunklevel_t lvl) const {
 228     return list_for_level(lvl)->num_chunks();
 229   }
 230 
 231   // Returns number of chunks for a given level.
 232   size_t committed_word_size_at_level(chunklevel_t lvl) const {
 233     return list_for_level(lvl)->committed_word_size();
 234   }
 235 
 236   // Returns reference to first chunk at this level, or NULL if sublist is empty.
 237   Metachunk* first_at_level(chunklevel_t lvl) const {
 238     return list_for_level(lvl)->first();
 239   }
 240 
 241   // Look for a chunk: starting at level, up to and including max_level,
 242   //  return the first chunk whose committed words >= min_committed_words.
 243   // Return NULL if no such chunk was found.
 244   Metachunk* search_chunk_ascending(chunklevel_t level, chunklevel_t max_level,
 245                                     size_t min_committed_words);
 246 
 247   // Look for a chunk: starting at level, down to (including) the root chunk level,
 248   // return the first chunk whose committed words >= min_committed_words.
 249   // Return NULL if no such chunk was found.
 250   Metachunk* search_chunk_descending(chunklevel_t level, size_t min_committed_words);
 251 
 252   // Returns total size in all lists (regardless of commit state of underlying memory)
 253   size_t word_size() const;
 254 
 255   // Returns total committed size in all lists
 256   size_t committed_word_size() const;
 257 
 258   // Returns number of chunks in all lists
 259   int num_chunks() const;
 260 
 261 #ifdef ASSERT
 262   bool contains(const Metachunk* c) const;
 263   void verify() const;
 264 #endif
 265 
 266   void print_on(outputStream* st) const;
 267 
 268 };
