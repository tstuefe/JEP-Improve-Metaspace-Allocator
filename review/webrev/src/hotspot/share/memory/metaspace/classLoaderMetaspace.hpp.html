  26 #ifndef SHARE_MEMORY_METASPACE_CLASSLOADERMETASPACE_HPP
  27 #define SHARE_MEMORY_METASPACE_CLASSLOADERMETASPACE_HPP
  28 
  29 #include "memory/allocation.hpp"
  30 #include "memory/metaspace/metaspaceEnums.hpp"
  31 
  32 class Mutex;
  33 
  34 namespace metaspace {
  35 
  36 class MetaspaceArena;
  37 struct clms_stats_t;
  38 
  39 class ClassLoaderMetaspace : public CHeapObj<mtClass> {
  40 
  41   // The CLD lock.
  42   Mutex* const _lock;
  43 
  44   const MetaspaceType _space_type;
  45 
  46   // Arena for allocations from non-class  metaspace
  47   //  (resp. for all allocations if -XX:-UseCompressedClassPointers).
  48   MetaspaceArena* _non_class_space_arena;
  49 
  50   // Arena for allocations from class space
  51   //  (NULL if -XX:-UseCompressedClassPointers).
  52   MetaspaceArena* _class_space_arena;
  53 
  54   Mutex* lock() const                             { return _lock; }
  55   MetaspaceArena* non_class_space_arena() const   { return _non_class_space_arena; }
  56   MetaspaceArena* class_space_arena() const       { return _class_space_arena; }
  57 
  58   MetaspaceArena* get_arena(bool is_class) {
  59     return is_class ? class_space_arena() : non_class_space_arena();
  60   }
  61 
  62   // Returns true if this class loader is of a type which will only ever load one class.
  63   bool is_micro() const {
  64     return _space_type == metaspace::ClassMirrorHolderMetaspaceType ||
  65            _space_type == metaspace::ReflectionMetaspaceType;
  66   }
  67 
  68 public:
  69 
  70   ClassLoaderMetaspace(Mutex* lock, MetaspaceType space_type);
  71 
  72   ~ClassLoaderMetaspace();
  73 
  74   MetaspaceType space_type() const { return _space_type; }
  75 
  76   // Allocate word_size words from Metaspace.
  77   MetaWord* allocate(size_t word_size, Metaspace::MetadataType mdType);
  78 
  79   // Attempt to expand the GC threshold to be good for at least another word_size words
  80   // and allocate. Returns NULL if failure. Used during Metaspace GC.
  81   MetaWord* expand_and_allocate(size_t word_size, Metaspace::MetadataType mdType);
  82 
  83   // Prematurely returns a metaspace allocation to the _block_freelists
  84   // because it is not needed anymore.
  85   void deallocate(MetaWord* ptr, size_t word_size, bool is_class);
  86 
  87   // Update statistics. This walks all in-use chunks.
  88   void add_to_statistics(clms_stats_t* out) const;
  89 
  90   DEBUG_ONLY(void verify() const;)
  91 
  92   // This only exists for JFR and jcmd VM.classloader_stats. We may want to
  93   //  change this. Capacity as a stat is of questionable use since it may
  94   //  contain committed and uncommitted areas. For now we do this to maintain
  95   //  backward compatibility with JFR.
  96   void calculate_jfr_stats(size_t* p_used_bytes, size_t* p_capacity_bytes) const;
  97 
  98 
  99 };
 100 
 101 } // namespace metaspace
 102 
 103 #endif // SHARE_MEMORY_METASPACE_CLASSLOADERMETASPACE_HPP
