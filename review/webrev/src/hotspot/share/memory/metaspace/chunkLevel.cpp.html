  35 chunklevel_t chunklevel::level_fitting_word_size(size_t word_size) {
  36 
  37   assert(chunklevel::MAX_CHUNK_WORD_SIZE >= word_size,
  38          SIZE_FORMAT " - too large allocation size.", word_size * BytesPerWord);
  39 
  40   // TODO: This can be done much better.
  41   chunklevel_t l = chunklevel::HIGHEST_CHUNK_LEVEL;
  42   while (l >= chunklevel::LOWEST_CHUNK_LEVEL &&
  43          word_size > chunklevel::word_size_for_level(l)) {
  44     l --;
  45   }
  46 
  47   return l;
  48 }
  49 
  50 void chunklevel::print_chunk_size(outputStream* st, chunklevel_t lvl) {
  51   if (chunklevel::is_valid_level(lvl)) {
  52     const size_t s = chunklevel::word_size_for_level(lvl) * BytesPerWord;
  53     if (s < 1 * M) {
  54       st->print("%3uk", (unsigned)(s / K));
  55     } else {
  56       st->print("%3um", (unsigned)(s / M));
  57     }
  58   } else {
  59     st->print("?-?");
  60   }
  61 
  62 }
