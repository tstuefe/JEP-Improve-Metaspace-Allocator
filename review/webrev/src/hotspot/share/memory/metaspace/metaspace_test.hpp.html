  50 // Wraps a MetaspaceTestArena with its own lock for testing purposes.
  51 class MetaspaceTestArena : public CHeapObj<mtInternal> {
  52 
  53   Mutex* const _lock;
  54   MetaspaceArena* const _arena;
  55 
  56 public:
  57 
  58   const MetaspaceArena* arena() const {
  59     return _arena;
  60   }
  61 
  62   MetaspaceTestArena(Mutex* lock, MetaspaceArena* arena);
  63   ~MetaspaceTestArena();
  64 
  65   MetaWord* allocate(size_t word_size);
  66   void deallocate(MetaWord* p, size_t word_size);
  67 
  68 };

  71 // Wraps an instance of a MetaspaceContext together with some side objects for easy use in test beds (whitebox, gtests)
  72 class MetaspaceTestContext : public CHeapObj<mtInternal> {
  73 
  74   const char* const _name;
  75   const size_t _reserve_limit;
  76   const size_t _commit_limit;
  77 
  78   MetaspaceContext* _context;
  79   CommitLimiter _commit_limiter;
  80   SizeAtomicCounter _used_words_counter;
  81 
  82 public:
  83 
  84   // Note: limit == 0 means unlimited
  85   // Reserve limit > 0 simulates a non-expandable VirtualSpaceList (like CompressedClassSpace)
  86   // Commit limit > 0 simulates a limit to max commitable space (like MaxMetaspaceSize)
  87   MetaspaceTestContext(const char* name, size_t commit_limit = 0, size_t reserve_limit = 0);
  88   ~MetaspaceTestContext();
  89 
  90   // Create an arena, feeding off this area.
  91   MetaspaceTestArena* create_arena(MetaspaceType type);
  92 
  93   void purge_area();
  94 
  95   // Accessors
  96   const CommitLimiter& commit_limiter() const { return _commit_limiter; }
  97   const VirtualSpaceList& vslist() const      { return *(_context->vslist()); }
  98   ChunkManager& cm()                          { return *(_context->cm()); }
  99 
 100   // Returns reserve- and commit limit we run the test with (in the real world,
 101   // these would be equivalent to CompressedClassSpaceSize resp MaxMetaspaceSize)
 102   size_t reserve_limit() const    { return _reserve_limit == 0 ? max_uintx : 0; }
 103   size_t commit_limit() const     { return _commit_limit == 0 ? max_uintx : 0; }
 104 
 105   // Convenience function to retrieve total committed/used words
 106   size_t used_words() const       { return _used_words_counter.get(); }
 107   size_t committed_words() const  { return _commit_limiter.committed_words(); }
 108 
 109   DEBUG_ONLY(void verify(bool slow = false) const;)
 110 
 111   void print_on(outputStream* st) const;
 112 
 113 };
