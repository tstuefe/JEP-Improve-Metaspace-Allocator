   3 import sun.hotspot.WhiteBox;
   4 
   5 import java.util.concurrent.atomic.AtomicLong;
   6 
   7 public class MetaspaceTestArena {
   8 
   9     long arena;
  10 
  11     final long allocationCeiling;
  12 
  13     // Number and word size of allocations
  14     long allocatedWords = 0;
  15     long numAllocated = 0;
  16     long deallocatedWords = 0;
  17     long numDeallocated = 0;
  18     long numAllocationFailures = 0;
  19 
  20     private synchronized boolean reachedCeiling() {
  21         return (allocatedWords - deallocatedWords) > allocationCeiling;
  22     }
  23 
  24     private synchronized void accountAllocation(long words) {
  25         numAllocated ++;
  26         allocatedWords += words;
  27     }
  28 
  29     private synchronized void accountDeallocation(long words) {
  30         numDeallocated ++;
  31         deallocatedWords += words;
  32     }
  33 
  34     MetaspaceTestArena(long arena0, long allocationCeiling) {
  35         this.allocationCeiling = allocationCeiling;
  36         this.arena = arena0;
  37     }
  38 
  39     public Allocation allocate(long words) {
  40         if (reachedCeiling()) {
  41             numAllocationFailures ++;
  42             return null;
  43         }
  44         WhiteBox wb = WhiteBox.getWhiteBox();
  45         long p = wb.allocateFromMetaspaceTestArena(arena, words);
  46         if (p == 0) {
  47             numAllocationFailures ++;
  48             return null;
  49         } else {
  50             accountAllocation(words);
  51         }
  52         return new Allocation(p, words);
  53     }
  54 
  55     public void deallocate(Allocation a) {
  56         WhiteBox wb = WhiteBox.getWhiteBox();
  57         wb.deallocateToMetaspaceTestArena(arena, a.p, a.word_size);
  58         accountDeallocation(a.word_size);
  59     }
  60 
  61     //// Convenience functions ////
  62 
  63     public Allocation allocate_expect_success(long words) {
  64         Allocation a = allocate(words);
  65         if (a.isNull()) {
  66             throw new RuntimeException("Allocation failed (" + words + ")");
  67         }
  68         return a;
  69     }
  70 
  71     public void allocate_expect_failure(long words) {
  72         Allocation a = allocate(words);
  73         if (!a.isNull()) {
  74             throw new RuntimeException("Allocation failed (" + words + ")");
  75         }
  76     }
  77 
  78     boolean isLive() {
  79         return arena != 0;
  80     }
  81 
  82 
  83     @Override
  84     public String toString() {
  85         return "arena=" + arena +
  86                 ", ceiling=" + allocationCeiling +
  87                 ", allocatedWords=" + allocatedWords +
  88                 ", numAllocated=" + numAllocated +
  89                 ", deallocatedWords=" + deallocatedWords +
  90                 ", numDeallocated=" + numDeallocated +
  91                 ", numAllocationFailures=" + numAllocationFailures +
  92                 '}';
  93     }

  95 }
