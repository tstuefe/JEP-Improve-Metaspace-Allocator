  25 import java.util.concurrent.CyclicBarrier;
  26 
  27 public class MetaspaceTestManyArenasManyThreads extends MetaspaceTestWithThreads {
  28 
  29     // Several threads allocate from a single arena.
  30     // This mimicks several threads loading classes via the same class loader.
  31 
  32     public MetaspaceTestManyArenasManyThreads(MetaspaceTestContext context, long testAllocationCeiling, int numThreads, int seconds) {
  33         super(context, testAllocationCeiling, numThreads, seconds);
  34     }
  35 
  36     public void runTest() throws Exception {
  37 
  38         CyclicBarrier gate = new CyclicBarrier(numThreads + 1);
  39 
  40         final long ceilingPerThread = testAllocationCeiling / numThreads;
  41 
  42         for (int i = 0; i < numThreads; i ++) {
  43             // Create n test threads, each one with its own allocator/arena pair
  44             MetaspaceTestArena arena = context.createArena(RandomHelper.fiftyfifty(), ceilingPerThread);
  45             RandomAllocator allocator = new RandomAllocator(arena);
  46             RandomAllocatorThread thread = new RandomAllocatorThread(gate, allocator, i);
  47             threads[i] = thread;
  48             thread.start();
  49         }
  50 
  51         gate.await();
  52 
  53         // while test is running, skim the arenas and kill any arena which is saturated (has started getting an
  54         // untoward number of allocation failures)
  55         long remainingMillis = seconds * 1000;
  56         while (remainingMillis > 0) {
  57 
  58             // Wait a bit
  59             int sleep_ms = 200;
  60             Thread.sleep(sleep_ms);
  61             remainingMillis -= sleep_ms;
  62 
  63             for (RandomAllocatorThread t: threads) {
  64                 if (t.allocator.arena.numAllocationFailures > 0) {
  65                     t.interrupt();
  66                     t.join();
  67                     context.destroyArena(t.allocator.arena);
  68 
  69                     // Create a new arena, allocator, then a new thread (note: do not pass in a start gate thjis time
  70                     // since we do not need to wait) and fire it up.
  71                     MetaspaceTestArena arena = context.createArena(RandomHelper.fiftyfifty(), ceilingPerThread);
  72                     RandomAllocator allocator = new RandomAllocator(arena);
  73                     RandomAllocatorThread t2 = new RandomAllocatorThread(null, allocator, t.id);
  74                     threads[t.id] = t2;
  75                     t2.start();
  76                 }
  77             }
  78 
  79         }
  80 
  81         // Stop all threads.
  82         stopAllThreads();
  83 
  84         context.updateTotals();
  85         System.out.println("  ## Finished: " + context);
  86 
  87         context.checkStatistics();
  88 
  89         // Destroy all arenas; then purge the space.
  90         destroyArenasAndPurgeSpace();
  91 
  92         context.destroy();
  93 
  94         context.updateTotals();
  95 
  96     }
  97 
  98 }
