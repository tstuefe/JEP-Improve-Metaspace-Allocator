1 import java.util.*;
   2 
   3 public class AllocationProfile {
   4 
   5     final String name;
   6 
   7     // Allocation word size spread
   8     public final long minimumSingleAllocationSize;
   9     public final long maximumSingleAllocationSize;
  10 
  11     // dealloc probability [0.0 .. 1.0]
  12     public final double randomDeallocProbability;
  13 
  14     public AllocationProfile(String name, long minimumSingleAllocationSize, long maximumSingleAllocationSize, double randomDeallocProbability) {
  15         this.minimumSingleAllocationSize = minimumSingleAllocationSize;
  16         this.maximumSingleAllocationSize = maximumSingleAllocationSize;
  17         this.randomDeallocProbability = randomDeallocProbability;
  18         this.name = name;
  19     }
  20 
  21     public long randomAllocationSize() {
  22         Random r = new Random();
  23         return r.nextInt((int)(maximumSingleAllocationSize - minimumSingleAllocationSize + 1)) + minimumSingleAllocationSize;
  24     }
  25 
  26 
  27     // Some standard profiles
  28     static final List<AllocationProfile> standardProfiles = new ArrayList<>();
  29 
  30     static {
  31         standardProfiles.add(new AllocationProfile("medium-range",1, 2048, 0.15));
  32         standardProfiles.add(new AllocationProfile("small-blocks",1, 512, 0.15));
  33         standardProfiles.add(new AllocationProfile("micro-blocks",1, 32, 0.15));
  34     }
  35 
  36     static AllocationProfile randomProfile() {
  37         return standardProfiles.get(RandomHelper.random().nextInt(standardProfiles.size()));
  38     }
  39 
  40     @Override
  41     public String toString() {
  42         return "MetaspaceTestAllocationProfile{" +
  43                 "name='" + name + '\'' +
  44                 ", minimumSingleAllocationSize=" + minimumSingleAllocationSize +
  45                 ", maximumSingleAllocationSize=" + maximumSingleAllocationSize +
  46                 ", randomDeallocProbability=" + randomDeallocProbability +
  47                 '}';
  48     }
  49 
  50 }
