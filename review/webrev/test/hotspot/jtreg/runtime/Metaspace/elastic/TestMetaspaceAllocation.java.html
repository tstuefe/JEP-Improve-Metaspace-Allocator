  56 public class TestMetaspaceAllocation {
  57 
  58     public static void main(String[] args) {
  59 
  60         MetaspaceTestContext context = new MetaspaceTestContext();
  61         MetaspaceTestArena arena1 = context.createArena(false, 1024 * 1024 * 4);
  62         MetaspaceTestArena arena2 = context.createArena(true,1024 * 1024 * 4);
  63 
  64         Allocation a1 = arena1.allocate(100);
  65         Allocation a2 = arena2.allocate(100);
  66 
  67         long used = context.usedWords();
  68         long committed = context.committedWords();
  69 
  70         System.out.println("used " + used + " committed " + committed);
  71 
  72         arena1.deallocate(a1);
  73 
  74         context.destroyArena(arena2);
  75         context.destroyArena(arena1);
  76 
  77         context.purge();
  78 
  79         context.destroy();
  80 
  81     }
  82 }
