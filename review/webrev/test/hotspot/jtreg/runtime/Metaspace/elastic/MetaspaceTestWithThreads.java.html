1 import java.util.Set;
   2 
   3 public class MetaspaceTestWithThreads {
   4 
   5     // The context to use.
   6     final MetaspaceTestContext context;
   7 
   8     // Total *word* size we allow for the test to allocation. The test may overshoot this a bit, but should not by much.
   9     final long testAllocationCeiling;
  10 
  11     // Number of parallel allocators
  12     final int numThreads;
  13 
  14     // Number of seconds for each test
  15     final int seconds;
  16 
  17     RandomAllocatorThread threads[];
  18 
  19     public MetaspaceTestWithThreads(MetaspaceTestContext context, long testAllocationCeiling, int numThreads, int seconds) {
  20         this.context = context;
  21         this.testAllocationCeiling = testAllocationCeiling;
  22         this.numThreads = numThreads;
  23         this.seconds = seconds;
  24         this.threads = new RandomAllocatorThread[numThreads];
  25     }
  26 
  27     protected void stopAllThreads() throws InterruptedException {
  28         // Stop all threads.
  29         for (Thread t: threads) {
  30             t.interrupt();
  31             t.join();
  32         }
  33     }
  34 
  35     void destroyArenasAndPurgeSpace() {
  36 
  37         for (RandomAllocatorThread t: threads) {
  38             if (t.allocator.arena.isLive()) {
  39                 context.destroyArena(t.allocator.arena);
  40             }
  41         }
  42 
  43         context.checkStatistics();
  44 
  45         // After deleting all arenas, we should have no committed space left: all arena chunks have been returned to
  46         // the freelist amd should have been maximally merged to a bunch of root chunks, which should be uncommitted
  47         // in one go.
  48         // Exception: if reclamation policy is none.
  49         if (Settings.settings().doesReclaim()) {
  50             if (context.committedWords() > 0) {
  51                 throw new RuntimeException("Expected no committed words after purging empty metaspace context (was: " + context.committedWords() + ")");
  52             }
  53         }
  54 
  55         context.purge();
  56 
  57         context.checkStatistics();
  58 
  59         // After purging - if all arenas had been deleted before - we should have no committed space left even in
  60         //   recmalation=none mode:
  61         // purging deletes all nodes with only free chunks, and in this case no node should still house in-use chunks,
  62         //  so all nodes would have been unmapped.
  63         // This is independent on reclamation policy. Only one exception: if the area was created with a reserve limit
  64         // (mimicking compressed class space), the underlying virtual space list cannot be purged.
  65         if (context.reserveLimit == 0) {
  66             if (context.committedWords() > 0) {
  67                 throw new RuntimeException("Expected no committed words after purging empty metaspace context (was: " + context.committedWords() + ")");
  68             }
  69         }
  70 
  71     }
  72 
  73     @Override
  74     public String toString() {
  75         return "commitLimit=" + context.commitLimit +
  76                 ", reserveLimit=" + context.reserveLimit +
  77                 ", testAllocationCeiling=" + testAllocationCeiling +
  78                 ", num_allocators=" + numThreads +
  79                 ", seconds=" + seconds;
  80     }
  81 
  82 
  83 }
