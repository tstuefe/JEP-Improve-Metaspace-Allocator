  25 #ifndef GTEST_METASPACE_METASPACE_TESTHELPER_HPP
  26 #define GTEST_METASPACE_METASPACE_TESTHELPER_HPP
  27 
  28 #include "memory/allocation.hpp"
  29 #include "metaspace/metaspaceTestsCommon.hpp"
  30 
  31 using namespace metaspace::chunklevel;
  32 
  33 class MetaspaceTestHelper : public metaspace::MetaspaceTestContext {
  34 
  35   int _num_chunks_allocated;
  36 
  37   void checked_alloc_chunk_0(Metachunk** p_return_value, chunklevel_t preferred_level,
  38                              chunklevel_t max_level, size_t min_committed_size);
  39 
  40 public:
  41 
  42   // Note: limit = 0 means unlimited
  43   MetaspaceTestHelper(size_t commit_limit = 0, size_t reserve_limit = 0);
  44 
  45   /////
  46 
  47   // Note: all test functions return void and return values are by pointer ref; this is awkward but otherwise we cannot
  48   // use gtest ASSERT macros inside those functions.
  49 
  50   // Allocate a chunk (you do not know if it will succeed).
  51   void alloc_chunk(Metachunk** p_return_value, chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  52     checked_alloc_chunk_0(p_return_value, preferred_level, max_level, min_committed_size);
  53   }
  54 
  55   // Allocate a chunk; do not expect success, but if it succeeds, test the chunk.
  56   void alloc_chunk(Metachunk** p_return_value, chunklevel_t level) {
  57     alloc_chunk(p_return_value, level, level, word_size_for_level(level));
  58   }
  59 
  60   // Allocate a chunk; it must succeed. Test the chunk.
  61   void alloc_chunk_expect_success(Metachunk** p_return_value, chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  62     checked_alloc_chunk_0(p_return_value, preferred_level, max_level, min_committed_size);
  63     ASSERT_NOT_NULL(*p_return_value);
  64   }
  65 
  66   // Allocate a chunk; it must succeed. Test the chunk.
  67   void alloc_chunk_expect_success(Metachunk** p_return_value, chunklevel_t level) {
  68     alloc_chunk_expect_success(p_return_value, level, level, word_size_for_level(level));
  69   }
  70 
  71   // Allocate a chunk but expect it to fail.
  72   void alloc_chunk_expect_failure(chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  73     Metachunk* c = NULL;
  74     checked_alloc_chunk_0(&c, preferred_level, max_level, min_committed_size);
  75     ASSERT_NULL(c);
  76   }
  77 
  78   // Allocate a chunk but expect it to fail.
  79   void alloc_chunk_expect_failure(chunklevel_t level) {
  80     return alloc_chunk_expect_failure(level, level, word_size_for_level(level));
  81   }
  82 
  83   /////
  84 
  85   void return_chunk(Metachunk* c);
  86 
  87   /////
  88 
  89   // Allocates from a chunk; also, fills allocated area with test pattern which will be tested with test_pattern().
  90   void allocate_from_chunk(MetaWord** p_return_value, Metachunk* c, size_t word_size);
  91 
  92   // Convenience function: allocate from chunk for when you don't care for the result pointer
  93   void allocate_from_chunk(Metachunk* c, size_t word_size) {
  94     MetaWord* dummy;
  95     allocate_from_chunk(&dummy, c, word_size);
  96   }
  97 
  98   // Test pattern established when allocating from the chunk with allocate_from_chunk_with_tests().
  99   void test_pattern(Metachunk* c, size_t word_size);
 100   void test_pattern(Metachunk* c) { test_pattern(c, c->used_words()); }
 101 
 102   void commit_chunk_with_test(Metachunk* c, size_t additional_size);
 103   void commit_chunk_expect_failure(Metachunk* c, size_t additional_size);
 104 
 105   void uncommit_chunk_with_test(Metachunk* c);
 106 
 107 
 108 };
 109 
 110 
 111 #endif // GTEST_METASPACE_METASPACE_TESTHELPER_HPP
