  35 void zap_range(MetaWord* p, size_t word_size) {
  36   for (MetaWord* pzap = p; pzap < p + word_size; pzap += os::vm_page_size() / BytesPerWord) {
  37     *pzap = (MetaWord)NOT_LP64(0xFEFEFEFE) LP64_ONLY(0xFEFEFEFEEFEFEFEFULL);
  38   }
  39 }

  43 // Writes a uniqe pattern to p
  44 void mark_address(MetaWord* p, uintx pattern) {
  45   MetaWord x = (MetaWord)((uintx) p ^ pattern);
  46   *p = x;
  47 }
  48 
  49 // checks pattern at address
  50 void check_marked_address(const MetaWord* p, uintx pattern) {
  51   MetaWord x = (MetaWord)((uintx) p ^ pattern);
  52   EXPECT_EQ(*p, x);
  53 }

  55 // "fill_range_with_pattern" fills a range of heap words with pointers to itself.
  56 //
  57 // The idea is to fill a memory range with a pattern which is both marked clearly to the caller
  58 // and cannot be moved without becoming invalid.
  59 //
  60 // The filled range can be checked with check_range_for_pattern. One also can only check
  61 // a sub range of the original range.
  62 void fill_range_with_pattern(MetaWord* p, size_t word_size, uintx pattern) {
  63   assert(word_size > 0 && p != NULL, "sanity");
  64   for (MetaWord* p2 = p; p2 < p + word_size; p2 ++) {
  65     mark_address(p2, pattern);
  66   }
  67 }
  68 
  69 void check_range_for_pattern(const MetaWord* p, size_t word_size, uintx pattern) {
  70   assert(p != NULL, "sanity");
  71   const MetaWord* p2 = p;
  72   while (p2 < p + word_size) {
  73     check_marked_address(p2, pattern);
  74     p2 ++;
  75   }
  76 }
  77 
  78 
  79 // Similar to fill_range_with_pattern, but only marks start and end. This is optimized for cases
  80 // where fill_range_with_pattern just is too slow.
  81 // Use check_marked_range to check the range. In contrast to check_range_for_pattern, only the original
  82 // range can be checked.
  83 void mark_range(MetaWord* p, size_t word_size, uintx pattern) {
  84   assert(word_size > 0 && p != NULL, "sanity");
  85   mark_address(p, pattern);
  86   mark_address(p + word_size - 1, pattern);
  87 }
  88 
  89 void check_marked_range(const MetaWord* p, size_t word_size, uintx pattern) {
  90   assert(word_size > 0 && p != NULL, "sanity");
  91   check_marked_address(p, pattern);
  92   check_marked_address(p + word_size - 1, pattern);
  93 }
  94 
  95 void mark_range(MetaWord* p, size_t word_size) {
  96   assert(word_size > 0 && p != NULL, "sanity");
  97   uintx pattern = (uintx)p2i(p);
  98   mark_range(p, word_size, pattern);
  99 }
 100 
 101 void check_marked_range(const MetaWord* p, size_t word_size) {
 102   uintx pattern = (uintx)p2i(p);
 103   check_marked_range(p, word_size, pattern);
 104 }
