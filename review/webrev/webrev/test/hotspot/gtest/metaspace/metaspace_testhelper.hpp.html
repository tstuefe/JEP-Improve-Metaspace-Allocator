  25 #ifndef GTEST_METASPACE_METASPACE_TESTHELPER_HPP
  26 #define GTEST_METASPACE_METASPACE_TESTHELPER_HPP
  27 
  28 #include "memory/allocation.hpp"
  29 #include "metaspace/metaspaceTestsCommon.hpp"
  30 
  31 using namespace metaspace::chunklevel;
  32 
  33 class MetaspaceTestHelper : public StackObj {
  34 
  35   ReservedSpace _rs;
  36   CommitLimiter _commit_limiter;
  37   VirtualSpaceList _vs_list;
  38   ChunkManager _cm;
  39   int _num_chunks_allocated;
  40 
  41   void checked_alloc_chunk_0(Metachunk** p_return_value, chunklevel_t preferred_level,
  42                              chunklevel_t max_level, size_t min_committed_size);
  43 
  44 public:
  45 
  46   // No reserve limit, and a commit limit.
  47   MetaspaceTestHelper(size_t commit_limit);
  48 
  49   // Reserve limit and commit limit.
  50   MetaspaceTestHelper(size_t reserve_limit, size_t commit_limit);
  51 
  52   // Default ctor should cause no limits to fire (within reason)
  53   MetaspaceTestHelper ();
  54 
  55   ~MetaspaceTestHelper();
  56 
  57   const CommitLimiter& commit_limiter() const { return _commit_limiter; }
  58   const VirtualSpaceList& vslist() const { return _vs_list; }
  59   ChunkManager& cm() { return _cm; }
  60 
  61   // Returns reserve- and commit limit we run the test with (in the real world,
  62   // these would be equivalent to CompressedClassSpaceSize resp MaxMetaspaceSize)
  63   size_t reserve_limit() const { return _rs.is_reserved() ? _rs.size() : max_uintx; }
  64   size_t commit_limit() const { return _commit_limiter.cap(); }
  65 
  66   /////
  67 
  68   // Note: all test functions return void and return values are by pointer ref; this is awkward but otherwise we cannot
  69   // use gtest ASSERT macros inside those functions.
  70 
  71   // Allocate a chunk (you do not know if it will succeed).
  72   void alloc_chunk(Metachunk** p_return_value, chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  73     checked_alloc_chunk_0(p_return_value, preferred_level, max_level, min_committed_size);
  74   }
  75 
  76   // Allocate a chunk; do not expect success, but if it succeeds, test the chunk.
  77   void alloc_chunk(Metachunk** p_return_value, chunklevel_t level) {
  78     alloc_chunk(p_return_value, level, level, word_size_for_level(level));
  79   }
  80 
  81   // Allocate a chunk; it must succeed. Test the chunk.
  82   void alloc_chunk_expect_success(Metachunk** p_return_value, chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  83     checked_alloc_chunk_0(p_return_value, preferred_level, max_level, min_committed_size);
  84     ASSERT_NOT_NULL(*p_return_value);
  85   }
  86 
  87   // Allocate a chunk; it must succeed. Test the chunk.
  88   void alloc_chunk_expect_success(Metachunk** p_return_value, chunklevel_t level) {
  89     alloc_chunk_expect_success(p_return_value, level, level, word_size_for_level(level));
  90   }
  91 
  92   // Allocate a chunk but expect it to fail.
  93   void alloc_chunk_expect_failure(chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  94     Metachunk* c = NULL;
  95     checked_alloc_chunk_0(&c, preferred_level, max_level, min_committed_size);
  96     ASSERT_NULL(c);
  97   }
  98 
  99   // Allocate a chunk but expect it to fail.
 100   void alloc_chunk_expect_failure(chunklevel_t level) {
 101     return alloc_chunk_expect_failure(level, level, word_size_for_level(level));
 102   }
 103 
 104   /////
 105 
 106   void return_chunk(Metachunk* c);
 107 
 108   /////
 109 
 110   // Allocates from a chunk; also, fills allocated area with test pattern which will be tested with test_pattern().
 111   void allocate_from_chunk(MetaWord** p_return_value, Metachunk* c, size_t word_size);
 112 
 113   // Convenience function: allocate from chunk for when you don't care for the result pointer
 114   void allocate_from_chunk(Metachunk* c, size_t word_size) {
 115     MetaWord* dummy;
 116     allocate_from_chunk(&dummy, c, word_size);
 117   }
 118 
 119   // Test pattern established when allocating from the chunk with allocate_from_chunk_with_tests().
 120   void test_pattern(Metachunk* c, size_t word_size);
 121   void test_pattern(Metachunk* c) { test_pattern(c, c->used_words()); }
 122 
 123   void commit_chunk_with_test(Metachunk* c, size_t additional_size);
 124   void commit_chunk_expect_failure(Metachunk* c, size_t additional_size);
 125 
 126   void uncommit_chunk_with_test(Metachunk* c);
 127 
 128   DEBUG_ONLY(void verify() const;)
 129 
 130 };
 131 
 132 
 133 #endif // GTEST_METASPACE_METASPACE_TESTHELPER_HPP
