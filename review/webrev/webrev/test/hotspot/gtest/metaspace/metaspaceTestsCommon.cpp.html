  38 size_t get_workingset_size() {
  39 #if defined(_WIN32)
  40   PROCESS_MEMORY_COUNTERS info;
  41   GetProcessMemoryInfo(GetCurrentProcess(), &info, sizeof(info));
  42   return (size_t)info.WorkingSetSize;
  43 #elif defined(LINUX)
  44   long result = 0L;
  45   FILE* f = fopen("/proc/self/statm", "r");
  46   if (f == NULL) {
  47     return 0;
  48   }
  49   // Second number in statm, in num of pages
  50   if (fscanf(f, "%*s%ld", &result) != 1 ) {
  51     fclose(f);
  52     return 0;
  53   }
  54   fclose(f);
  55   return (size_t)result * (size_t)os::vm_page_size();
  56 #else
  57   return 0L;
  58 #endif
  59 }

  63 void zap_range(MetaWord* p, size_t word_size) {
  64   for (MetaWord* pzap = p; pzap < p + word_size; pzap += os::vm_page_size() / BytesPerWord) {
  65     *pzap = (MetaWord)NOT_LP64(0xFEFEFEFE) LP64_ONLY(0xFEFEFEFEEFEFEFEFULL);
  66   }
  67 }

  71 // Writes a uniqe pattern to p
  72 void mark_address(MetaWord* p, uintx pattern) {
  73   MetaWord x = (MetaWord)((uintx) p ^ pattern);
  74   *p = x;
  75 }
  76 
  77 // checks pattern at address
  78 void check_marked_address(const MetaWord* p, uintx pattern) {
  79   MetaWord x = (MetaWord)((uintx) p ^ pattern);
  80   EXPECT_EQ(*p, x);
  81 }

  83 // "fill_range_with_pattern" fills a range of heap words with pointers to itself.
  84 //
  85 // The idea is to fill a memory range with a pattern which is both marked clearly to the caller
  86 // and cannot be moved without becoming invalid.
  87 //
  88 // The filled range can be checked with check_range_for_pattern. One also can only check
  89 // a sub range of the original range.
  90 void fill_range_with_pattern(MetaWord* p, size_t word_size, uintx pattern) {
  91   assert(word_size > 0 && p != NULL, "sanity");
  92   for (MetaWord* p2 = p; p2 < p + word_size; p2 ++) {
  93     mark_address(p2, pattern);
  94   }
  95 }
  96 
  97 void check_range_for_pattern(const MetaWord* p, size_t word_size, uintx pattern) {
  98   assert(p != NULL, "sanity");
  99   const MetaWord* p2 = p;
 100   while (p2 < p + word_size) {
 101     check_marked_address(p2, pattern);
 102     p2 ++;
 103   }
 104 }

 107 // Similar to fill_range_with_pattern, but only marks start and end. This is optimized for cases
 108 // where fill_range_with_pattern just is too slow.
 109 // Use check_marked_range to check the range. In contrast to check_range_for_pattern, only the original
 110 // range can be checked.
 111 void mark_range(MetaWord* p, size_t word_size, uintx pattern) {
 112   assert(word_size > 0 && p != NULL, "sanity");
 113   mark_address(p, pattern);
 114   mark_address(p + word_size - 1, pattern);
 115 }
 116 
 117 void check_marked_range(const MetaWord* p, size_t word_size, uintx pattern) {
 118   assert(word_size > 0 && p != NULL, "sanity");
 119   check_marked_address(p, pattern);
 120   check_marked_address(p + word_size - 1, pattern);
 121 }
 122 
 123 void mark_range(MetaWord* p, size_t word_size) {
 124   assert(word_size > 0 && p != NULL, "sanity");
 125   uintx pattern = (uintx)p2i(p);
 126   mark_range(p, word_size, pattern);
 127 }
 128 
 129 void check_marked_range(const MetaWord* p, size_t word_size) {
 130   uintx pattern = (uintx)p2i(p);
 131   check_marked_range(p, word_size, pattern);
 132 }
