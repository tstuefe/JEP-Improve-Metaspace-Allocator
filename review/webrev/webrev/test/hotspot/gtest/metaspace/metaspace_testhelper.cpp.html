  67 void MetaspaceTestHelper::checked_alloc_chunk_0(Metachunk** p_return_value, chunklevel_t preferred_level, chunklevel_t max_level,
  68                                                       size_t min_committed_size) {
  69 
  70   *p_return_value = NULL;
  71 
  72   Metachunk* c = cm().get_chunk(preferred_level, max_level, min_committed_size);
  73 
  74   if (c != NULL) {
  75 
  76     ASSERT_LE(c->level(), max_level);
  77     ASSERT_GE(c->level(), preferred_level);
  78     ASSERT_GE(c->committed_words(), min_committed_size);
  79     ASSERT_EQ(c->committed_words(), c->free_below_committed_words());
  80     ASSERT_EQ(c->used_words(), (size_t)0);
  81     ASSERT_TRUE(c->is_in_use());
  82     ASSERT_FALSE(c->is_free());
  83     ASSERT_FALSE(c->is_dead());
  84     ASSERT_NULL(c->next());
  85     ASSERT_NULL(c->prev());
  86     if (c->level() == HIGHEST_CHUNK_LEVEL) {
  87       ASSERT_TRUE(c->is_leaf_chunk());
  88     } else {
  89       ASSERT_FALSE(c->is_leaf_chunk());
  90     }
  91     if (c->level() == LOWEST_CHUNK_LEVEL) {
  92       ASSERT_TRUE(c->is_root_chunk());
  93     } else {
  94       ASSERT_FALSE(c->is_root_chunk());
  95     }
  96     if (_num_chunks_allocated == 0) { // First chunk? We can make more assumptions
  97       ASSERT_EQ(c->level(), preferred_level);
  98       // Needs lock EXPECT_NULL(c->next_in_vs());
  99       // Needs lock EXPECT_NULL(c->prev_in_vs());
 100       ASSERT_TRUE(c->is_root_chunk() || c->is_leader());
 101     }
 102     _num_chunks_allocated ++;
 103 
 104   }
 105 
 106   DEBUG_ONLY(verify();)
 107 
 108   *p_return_value = c;
 109 
 110 }

 112 // Test pattern established when allocating from the chunk with allocate_from_chunk_with_tests().
 113 void MetaspaceTestHelper::test_pattern(Metachunk* c, size_t word_size) {
 114   check_range_for_pattern(c->base(), word_size, (uintx)c);
 115 }

 117 void MetaspaceTestHelper::return_chunk(Metachunk* c) {
 118   test_pattern(c);
 119   DEBUG_ONLY(verify();)
 120   c->set_in_use(); // Forestall assert in cm
 121   _cm.return_chunk(c);

 124  void MetaspaceTestHelper::allocate_from_chunk(MetaWord** p_return_value, Metachunk* c, size_t word_size) {
 125 
 126   size_t used_before = c->used_words();
 127   size_t free_before = c->free_words();
 128   size_t free_below_committed_before = c->free_below_committed_words();
 129   const MetaWord* top_before = c->top();
 130 
 131   MetaWord* p = c->allocate(word_size);
 132   EXPECT_NOT_NULL(p);
 133   EXPECT_EQ(c->used_words(), used_before + word_size);
 134   EXPECT_EQ(c->free_words(), free_before - word_size);
 135   EXPECT_EQ(c->free_below_committed_words(), free_below_committed_before - word_size);
 136   EXPECT_EQ(c->top(), top_before + word_size);
 137 
 138   // Old content should be preserved
 139   test_pattern(c, used_before);
 140 
 141   // Fill newly allocated range too
 142   fill_range_with_pattern(p, word_size, (uintx)c);
 143 
 144   *p_return_value = p;
 145 }

 147 void MetaspaceTestHelper::commit_chunk_with_test(Metachunk* c, size_t additional_size) {
 148 
 149   size_t used_before = c->used_words();
 150   size_t free_before = c->free_words();
 151   const MetaWord* top_before = c->top();
 152 
 153   c->set_in_use();
 154   bool b = c->ensure_committed_additional(additional_size);
 155   EXPECT_TRUE(b);
 156 
 157   // We should have enough committed size now
 158   EXPECT_GE(c->free_below_committed_words(), additional_size);
 159 
 160   // used, free, top should be unchanged.
 161   EXPECT_EQ(c->used_words(), used_before);
 162   EXPECT_EQ(c->free_words(), free_before);
 163   EXPECT_EQ(c->top(), top_before);
 164 
 165   test_pattern(c, used_before);
 166 
 167 }

 169 void MetaspaceTestHelper::commit_chunk_expect_failure(Metachunk* c, size_t additional_size) {
 170 
 171   size_t used_before = c->used_words();
 172   size_t free_before = c->free_words();
 173   size_t free_below_committed_before = c->free_below_committed_words();
 174   const MetaWord* top_before = c->top();
 175 
 176   c->set_in_use();
 177   bool b = c->ensure_committed_additional(additional_size);
 178   EXPECT_FALSE(b);
 179 
 180   // Nothing should have changed
 181   EXPECT_EQ(c->used_words(), used_before);
 182   EXPECT_EQ(c->free_words(), free_before);
 183   EXPECT_EQ(c->free_below_committed_words(), free_below_committed_before);
 184   EXPECT_EQ(c->top(), top_before);
 185 
 186   test_pattern(c, used_before);
 187 
 188 }

 190 void MetaspaceTestHelper::uncommit_chunk_with_test(Metachunk* c) {
 191   if (c->word_size() >= Settings::commit_granule_words()) {
 192     c->set_free();  // Forestall assert in uncommit
 193     c->reset_used_words();
 194     c->uncommit();
 195 
 196     EXPECT_EQ(c->free_below_committed_words(), (size_t)0);
 197     EXPECT_EQ(c->used_words(), (size_t)0);
 198     EXPECT_EQ(c->free_words(), c->word_size());
 199     EXPECT_EQ(c->top(), c->base());
 200     EXPECT_TRUE(c->is_fully_uncommitted());
 201   }
 202 }
