  25 #ifndef SHARE_MEMORY_METADATAFACTORY_HPP
  26 #define SHARE_MEMORY_METADATAFACTORY_HPP
  27 
  28 #include "classfile/classLoaderData.hpp"
  29 #include "memory/metaspace/classLoaderMetaspace.hpp"
  30 #include "memory/metaspace/metaspaceEnums.hpp"
  31 #include "oops/array.hpp"
  32 #include "utilities/exceptions.hpp"
  33 #include "utilities/globalDefinitions.hpp"
  34 
  35 class MetadataFactory : AllStatic {
  36  public:
  37   template <typename T>
  38   static Array<T>* new_array(ClassLoaderData* loader_data, int length, TRAPS) {
  39     // The "true" argument is because all metadata arrays are read only when
  40     // dumped to the shared archive
  41     return new (loader_data, length, THREAD) Array<T>(length);
  42   }
  43 
  44   template <typename T>
  45   static Array<T>* new_array(ClassLoaderData* loader_data, int length, T value, TRAPS) {
  46     Array<T>* array = new_array<T>(loader_data, length, CHECK_NULL);
  47     for (int i = 0; i < length; i++) {
  48       array->at_put(i, value);
  49     }
  50     return array;
  51   }
  52 
  53   template <typename T>
  54   static void free_array(ClassLoaderData* loader_data, Array<T>* data) {
  55     if (data != NULL) {
  56       assert(loader_data != NULL, "shouldn't pass null");
  57       assert(!data->is_shared(), "cannot deallocate array in shared spaces");
  58       int size = data->size();
  59       loader_data->metaspace_non_null()->deallocate((MetaWord*)data, size, false);
  60     }
  61   }
  62 
  63   // Deallocation method for metadata
  64   template <class T>
  65   static void free_metadata(ClassLoaderData* loader_data, T md) {
  66     if (md != NULL) {
  67       assert(loader_data != NULL, "shouldn't pass null");
  68       int size = md->size();
  69       // Call metadata's deallocate function which will call deallocate fields
  70       assert(!md->on_stack(), "can't deallocate things on stack");
  71       assert(!md->is_shared(), "cannot deallocate if in shared spaces");
  72       md->deallocate_contents(loader_data);
  73       loader_data->metaspace_non_null()->deallocate((MetaWord*)md, size, md->is_klass());
  74     }
  75   }
  76 };
  77 
  78 #endif // SHARE_MEMORY_METADATAFACTORY_HPP
