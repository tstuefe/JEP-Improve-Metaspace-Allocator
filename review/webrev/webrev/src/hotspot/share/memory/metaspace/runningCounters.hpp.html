  26 #ifndef SHARE_MEMORY_METASPACE_RUNNINGCOUNTERS_HPP
  27 #define SHARE_MEMORY_METASPACE_RUNNINGCOUNTERS_HPP
  28 
  29 #include "memory/allocation.hpp"
  30 #include "memory/metaspace/counter.hpp"
  31 
  32 namespace metaspace {
  33 
  34 class ClassLoaderMetaspace;
  35 
  36 // These are running counters for some basic Metaspace statistics.
  37 // Their value can be obtained quickly without locking.
  38 
  39 class RunningCounters : public AllStatic {
  40 
  41   friend class ClassLoaderMetaspace;
  42 
  43   // ---- in-use chunks ----
  44 
  45   // Used space, in words.
  46   // (Note that the used counter is on the hot path of Metaspace allocation.
  47   //  Do we really need it? We may get by with capacity only and get more details
  48   //  with get_statistics_slow().)
  49   static SizeAtomicCounter _used_class_counter;
  50   static SizeAtomicCounter _used_nonclass_counter;
  51 
  52 public:
  53 
  54   // ---- virtual memory -----
  55 
  56   // Return reserved size, in words, for Metaspace
  57   static size_t reserved_words();
  58   static size_t reserved_words_class();
  59   static size_t reserved_words_nonclass();
  60 
  61   // Return total committed size, in words, for Metaspace
  62   static size_t committed_words();
  63   static size_t committed_words_class();
  64   static size_t committed_words_nonclass();
  65 
  66 
  67   // ---- used chunks -----
  68 
  69   // Returns size, in words, used for metadata.
  70   static size_t used_words();
  71   static size_t used_words_class();
  72   static size_t used_words_nonclass();
  73 
  74   // ---- free chunks -----
  75 
  76   // Returns size, in words, of all chunks in all freelists.
  77   static size_t free_chunks_words();
  78   static size_t free_chunks_words_class();
  79   static size_t free_chunks_words_nonclass();
  80 
  81   // Direct access to the counters.
  82   static SizeAtomicCounter* used_nonclass_counter()     { return &_used_nonclass_counter; }
  83   static SizeAtomicCounter* used_class_counter()        { return &_used_class_counter; }
  84 
  85 };
  86 
  87 } // namespace metaspace
  88 
  89 #endif // SHARE_MEMORY_METASPACE_RUNNINGCOUNTERS_HPP
