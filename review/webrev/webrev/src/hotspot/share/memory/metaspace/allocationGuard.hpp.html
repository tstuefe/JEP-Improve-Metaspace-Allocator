  35 // In Debug builds, Metadata in Metaspace can be optionally guarded - enclosed in canaries -
  36 // to detect memory overwriters. This is switched on by "-XX:+MetaspaceGuardAllocations".
  37 //
  38 // These canaries are periodically checked, e.g. when the Metaspace is purged in a context
  39 // of a GC.
  40 
  41 namespace metaspace {
  42 
  43 #ifdef ASSERT
  44 
  45 struct prefix_t {
  46   uintx mark;
  47   size_t word_size;       // raw word size including prefix
  48   // MetaWord payload [0];   // varsized (but unfortunately not all our compilers understand that)
  49 };
  50 
  51 // The prefix structure must be aligned to MetaWord size.
  52 STATIC_ASSERT((sizeof(prefix_t) & WordAlignmentMask) == 0);
  53 
  54 inline prefix_t* prefix_from_payload(MetaWord* p) {
  55   return (prefix_t*)((address)p - sizeof(prefix_t));
  56 }
  57 
  58 inline MetaWord* payload_from_prefix(prefix_t* pp) {
  59   // return pp->payload;
  60   return (MetaWord*)((address)pp + sizeof(prefix_t));
  61 }
  62 
  63 inline size_t prefix_size() {
  64   return sizeof(prefix_t);
  65 }
  66 
  67 #define EYECATCHER NOT_LP64(0x77698465) LP64_ONLY(0x7769846577698465ULL) // "META" resp "METAMETA"
  68 
  69 // Given a pointer to a memory area, establish the prefix at the start of that area and
  70 // return the starting pointer to the payload.
  71 inline MetaWord* establish_prefix(MetaWord* p_raw, size_t raw_word_size) {
  72   prefix_t* pp = (prefix_t*)p_raw;
  73   pp->mark = EYECATCHER;
  74   pp->word_size = raw_word_size;
  75   return payload_from_prefix(pp);
  76 }
  77 
  78 inline void check_prefix(const prefix_t* pp) {
  79   assert(pp->mark == EYECATCHER, "corrupt block at " PTR_FORMAT ".", p2i(pp));
  80   assert(pp->word_size > 0 && pp->word_size < chunklevel::MAX_CHUNK_WORD_SIZE,
  81          "Invalid size " SIZE_FORMAT " in block at " PTR_FORMAT ".", pp->word_size, p2i(pp));
  82 }
  83 
  84 #endif
  85 
  86 } // namespace metaspace
  87 
  88 #endif // SHARE_MEMORY_METASPACE_ALLOCATION_GUARD_HPP
