  34 MetaspaceDCmd::MetaspaceDCmd(outputStream* output, bool heap)
  35   : DCmdWithParser(output, heap)
  36   , _basic("basic", "Prints a basic summary (does not need a safepoint).", "BOOLEAN", false, "false")
  37   , _show_loaders("show-loaders", "Shows usage by class loader.", "BOOLEAN", false, "false")
  38   , _by_spacetype("by-spacetype", "Break down numbers by loader type.", "BOOLEAN", false, "false")
  39   , _by_chunktype("by-chunktype", "Break down numbers by chunk type.", "BOOLEAN", false, "false")
  40   , _show_vslist("vslist", "Shows details about the underlying virtual space.", "BOOLEAN", false, "false")
  41   , _scale("scale", "Memory usage in which to scale. Valid values are: 1, KB, MB or GB (fixed scale) "
  42            "or \"dynamic\" for a dynamically choosen scale.",
  43            "STRING", false, "dynamic")
  44   , _show_classes("show-classes", "If show-loaders is set, shows loaded classes for each loader.", "BOOLEAN", false, "false")
  45 {
  46   _dcmdparser.add_dcmd_option(&_basic);
  47   _dcmdparser.add_dcmd_option(&_show_loaders);
  48   _dcmdparser.add_dcmd_option(&_show_classes);
  49   _dcmdparser.add_dcmd_option(&_by_chunktype);
  50   _dcmdparser.add_dcmd_option(&_by_spacetype);
  51   _dcmdparser.add_dcmd_option(&_show_vslist);
  52   _dcmdparser.add_dcmd_option(&_scale);
  53 }
  54 
  55 int MetaspaceDCmd::num_arguments() {
  56   ResourceMark rm;
  57   MetaspaceDCmd* dcmd = new MetaspaceDCmd(NULL, false);
  58   if (dcmd != NULL) {
  59     DCmdMark mark(dcmd);
  60     return dcmd->_dcmdparser.num_arguments();
  61   } else {
  62     return 0;
  63   }
  64 }
  65 
  66 void MetaspaceDCmd::execute(DCmdSource source, TRAPS) {
  67   // Parse scale value.
  68   const char* scale_value = _scale.value();
  69   size_t scale = 0;
  70   if (scale_value != NULL) {
  71     if (strcasecmp("dynamic", scale_value) == 0) {
  72       scale = 0;
  73     } else {
  74       scale = NMT_ONLY(NMTUtil::scale_from_name(scale_value)) NOT_NMT(0);
  75       if (scale == 0) {
  76         output()->print_cr("Invalid scale: \"%s\". Will use dynamic scaling.", scale_value);
  77       }
  78     }
  79   }
  80   if (_basic.value() == true) {
  81     if (_show_loaders.value() || _by_chunktype.value() || _by_spacetype.value() ||
  82         _show_vslist.value()) {
  83       // Basic mode. Just print essentials. Does not need to be at a safepoint.
  84       output()->print_cr("In basic mode, additional arguments are ignored.");
  85     }
  86     MetaspaceUtils::print_basic_report(output(), scale);
  87   } else {
  88     // Full mode. Requires safepoint.
  89     int flags = 0;
  90     if (_show_loaders.value())         flags |= MetaspaceReporter::rf_show_loaders;
  91     if (_show_classes.value())         flags |= MetaspaceReporter::rf_show_classes;
  92     if (_by_chunktype.value())         flags |= MetaspaceReporter::rf_break_down_by_chunktype;
  93     if (_by_spacetype.value())         flags |= MetaspaceReporter::rf_break_down_by_spacetype;
  94     if (_show_vslist.value())          flags |= MetaspaceReporter::rf_show_vslist;
  95     VM_PrintMetadata op(output(), scale, flags);
  96     VMThread::execute(&op);
  97   }
  98 }
