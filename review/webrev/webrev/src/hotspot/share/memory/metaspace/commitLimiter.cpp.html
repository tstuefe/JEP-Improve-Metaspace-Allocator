  35 // Returns the size, in words, by which we may expand the metaspace committed area without:
  36 // - _cap == 0: hitting GC threshold or the MaxMetaspaceSize
  37 // - _cap > 0: hitting cap (this is just for testing purposes)
  38 size_t CommitLimiter::possible_expansion_words() const {
  39 
  40   if (_cap > 0) { // Testing.
  41     assert(_cnt.get() <= _cap, "Beyond limit?");
  42     return _cap - _cnt.get();
  43   }
  44 
  45   assert(_cnt.get() * BytesPerWord <= MaxMetaspaceSize, "Beyond limit?");
  46   const size_t words_left_below_max = MaxMetaspaceSize / BytesPerWord - _cnt.get();
  47 
  48   const size_t words_left_below_gc_threshold = MetaspaceGC::allowed_expansion();
  49 
  50   return MIN2(words_left_below_max, words_left_below_gc_threshold);
  51 
  52 }
  53 
  54 static CommitLimiter g_global_limiter(0);
  55 
  56 // Returns the global metaspace commit counter
  57 CommitLimiter* CommitLimiter::globalLimiter() {
  58   return &g_global_limiter;
  59 }
