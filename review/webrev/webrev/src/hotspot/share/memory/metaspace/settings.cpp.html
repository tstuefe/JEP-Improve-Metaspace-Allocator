  43 size_t Settings::_committed_words_on_fresh_chunks = 0;
  44 
  45 bool Settings::_uncommit_on_return = false;
  46 size_t Settings::_uncommit_on_return_min_word_size = 0;
  47 
  48 bool Settings::_delete_nodes_on_purge = false;
  49 bool Settings::_uncommit_on_purge = false;

  56 void Settings::ergo_initialize() {
  57 
  58   if (strcmp(MetaspaceReclaimPolicy, "none") == 0) {
  59 
  60     log_info(metaspace)("Initialized with strategy: no reclaim.");
  61 
  62     _commit_granule_bytes = MAX2((size_t)os::vm_page_size(), 64 * K);
  63     _commit_granule_words = _commit_granule_bytes / BytesPerWord;
  64 
  65     _newborn_root_chunks_are_fully_committed = true;
  66 
  67     _committed_words_on_fresh_chunks = chunklevel::MAX_CHUNK_WORD_SIZE;
  68 
  69     _uncommit_on_return = false;
  70     _uncommit_on_return_min_word_size = 3; // does not matter; should not be used resp. assert when used.
  71 
  72     _delete_nodes_on_purge = false;
  73     _uncommit_on_purge = false;
  74     _uncommit_on_purge_min_word_size = 3; // does not matter; should not be used resp. assert when used.
  75 
  76   } else if (strcmp(MetaspaceReclaimPolicy, "aggressive") == 0) {
  77 
  78     log_info(metaspace)("Initialized with strategy: aggressive reclaim.");
  79 
  80     // Set the granule size rather small; may increase
  81     // mapping fragmentation but also increase chance to uncommit.
  82     _commit_granule_bytes = MAX2((size_t)os::vm_page_size(), 16 * K);
  83     _commit_granule_words = _commit_granule_bytes / BytesPerWord;
  84 
  85     _newborn_root_chunks_are_fully_committed = false;
  86 
  87     // When handing out fresh chunks, only commit the minimum sensible amount (0 would be possible
  88     // but not make sense since the chunk is immediately used for allocation after being handed out, so the
  89     // first granule would be committed right away anyway).
  90     _committed_words_on_fresh_chunks = _commit_granule_words;
  91 
  92     _uncommit_on_return = true;
  93     _uncommit_on_return_min_word_size = _commit_granule_words;
  94 
  95     _delete_nodes_on_purge = true;
  96     _uncommit_on_purge = true;
  97     _uncommit_on_purge_min_word_size = _commit_granule_words; // does not matter; should not be used resp. assert when used.
  98 
  99   } else if (strcmp(MetaspaceReclaimPolicy, "balanced") == 0) {
 100 
 101     log_info(metaspace)("Initialized with strategy: balanced reclaim.");
 102 
 103     _commit_granule_bytes = MAX2((size_t)os::vm_page_size(), 64 * K);
 104     _commit_granule_words = _commit_granule_bytes / BytesPerWord;
 105 
 106     _newborn_root_chunks_are_fully_committed = false;
 107 
 108     // When handing out fresh chunks, only commit the minimum sensible amount (0 would be possible
 109     // but not make sense since the chunk is immediately used for allocation after being handed out, so the
 110     // first granule would be committed right away anyway).
 111     _committed_words_on_fresh_chunks = _commit_granule_words;
 112 
 113     _uncommit_on_return = true;
 114     _uncommit_on_return_min_word_size = _commit_granule_words;
 115 
 116     _delete_nodes_on_purge = true;
 117     _uncommit_on_purge = true;
 118     _uncommit_on_purge_min_word_size = _commit_granule_words;
 119 
 120   } else {
 121 
 122     vm_exit_during_initialization("Invalid value for MetaspaceReclaimPolicy: \"%s\".", MetaspaceReclaimPolicy);
 123 
 124   }
 125 
 126   // Sanity checks.
 127   guarantee(commit_granule_words() <= chunklevel::MAX_CHUNK_WORD_SIZE, "Too large granule size");
 128   guarantee(is_power_of_2(commit_granule_words()), "granule size must be a power of 2");
 129 
 130 #ifdef ASSERT
 131   // Off for release builds, and by default for debug builds, but can be switched on manually to aid
 132   // error analysis.
 133   _use_allocation_guard = MetaspaceGuardAllocations;
 134 
 135   // Deallocations can be manually switched off to aid error analysis, since this removes one layer of complexity
 136   //  from allocation.
 137   _handle_deallocations = MetaspaceHandleDeallocations;
 138 
 139   // We also switch it off automatically if we use allocation guards. This is to keep prefix handling in SpaceManager simple.
 140   if (_use_allocation_guard) {
 141     _handle_deallocations = false;
 142   }
 143 #endif
 144 
 145   LogStream ls(Log(metaspace)::info());
 146   Settings::print_on(&ls);
 147 
 148 }
 149 
 150 void Settings::print_on(outputStream* st) {
 151 
 152   st->print_cr(" - commit_granule_bytes: " SIZE_FORMAT ".", commit_granule_bytes());
 153   st->print_cr(" - commit_granule_words: " SIZE_FORMAT ".", commit_granule_words());
 154 
 155   st->print_cr(" - newborn_root_chunks_are_fully_committed: %d.", (int)newborn_root_chunks_are_fully_committed());
 156   st->print_cr(" - committed_words_on_fresh_chunks: " SIZE_FORMAT ".", committed_words_on_fresh_chunks());
 157 
 158   st->print_cr(" - virtual_space_node_default_size: " SIZE_FORMAT ".", virtual_space_node_default_word_size());
 159   st->print_cr(" - allocation_from_dictionary_limit: " SIZE_FORMAT ".", allocation_from_dictionary_limit());
 160 
 161   st->print_cr(" - enlarge_chunks_in_place: %d.", (int)enlarge_chunks_in_place());
 162   st->print_cr(" - enlarge_chunks_in_place_max_word_size: " SIZE_FORMAT ".", enlarge_chunks_in_place_max_word_size());
 163 
 164   st->print_cr(" - uncommit_on_return: %d.", (int)uncommit_on_return());
 165   st->print_cr(" - uncommit_on_return_min_word_size: " SIZE_FORMAT ".", uncommit_on_return_min_word_size());
 166 
 167   st->print_cr(" - delete_nodes_on_purge: %d.", (int)delete_nodes_on_purge());
 168 
 169   st->print_cr(" - uncommit_on_purge: %d.", (int)uncommit_on_purge());
 170   st->print_cr(" - uncommit_on_purge_min_word_size: " SIZE_FORMAT ".", uncommit_on_purge_min_word_size());
 171 
 172   st->print_cr(" - use_allocation_guard: %d.", (int)use_allocation_guard());
 173   st->print_cr(" - handle_deallocations: %d.", (int)handle_deallocations());
 174 
 175 }
