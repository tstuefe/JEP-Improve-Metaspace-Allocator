  37 namespace metaspace {

  39 // TODO these are too useful to hide them in debug builds: those cheap and useful enough
  40 // should be available in release too.
  41 
  42 // A number of counters for internal metaspace statistics; Only active in debug.
  43 class InternalStats : public AllStatic {
  44 
  45   // Note: all counters which are modified on the classloader local allocation path
  46   //   (not under ExpandLock protection) have to be atomic.
  47 
  48 #define ALL_MY_COUNTERS(x, x_atomic)                \
  49                                                     \
  50   /* Number of allocations. */                      \
  51   x_atomic(num_allocs)                              \
  52                                                     \
  53   /* Number of deallocations (external) */          \
  54   x_atomic(num_deallocs)                            \
  55   /* Number of times an allocation was satisfied */ \
  56   /*  from deallocated blocks. */                   \
  57   x_atomic(num_allocs_from_deallocated_blocks)      \
  58                                                     \
  59   /* Number of times an allocation failed */        \
  60   /*  because the chunk was too small. */           \
  61         x_atomic(num_allocs_failed_chunk_too_small)       \
  62                                                     \
  63   /* Number of times an allocation failed */        \
  64   /*  because we hit a limit. */                    \
  65   x_atomic(num_allocs_failed_limit)                 \
  66                                                     \
  67   /* Number of times a ClassLoaderMetaspace was */  \
  68   /*  born... */                                    \
  69   x(num_metaspace_births)                           \
  70   /* ... and died. */                               \
  71   x(num_metaspace_deaths)                           \
  72                                                     \
  73         /* Number of times a ClassLoaderMetaspace was */  \
  74         /*  born for a hidden or anonymous class... */    \
  75         x(num_anon_cld_births)                            \
  76         /* ... and died. */                               \
  77   x(num_anon_cld_deaths)                            \
  78                                                     \
  79   /* Number of times VirtualSpaceNode were */       \
  80   /*  created...  */                                \
  81   x(num_vsnodes_created)                            \
  82   /* ... and purged. */                             \
  83   x(num_vsnodes_destroyed)                          \
  84                                                     \
  85   /* Number of times we committed space. */         \
  86   x(num_space_committed)                            \
  87   /* Number of times we uncommitted space. */       \
  88   x(num_space_uncommitted)                          \
  89                                                     \
  90   /* Number of times a chunk was returned to the */ \
  91   /*  freelist (external only). */                  \
  92   x(num_chunks_returned_to_freelist)                \
  93   /* Number of times a chunk was taken from */      \
  94   /*  freelist (external only) */                   \
  95   x(num_chunks_taken_from_freelist)                 \
  96                                                     \
  97   /* Number of successful chunk merges */           \
  98   x(num_chunk_merges)                               \
  99   /* Number of chunks removed from freelist as */   \
 100   /* result of a merge operation */                 \
 101   x(num_chunks_removed_from_freelist_due_to_merge)  \
 102                                                     \
 103   /* Number of chunk splits */                      \
 104   x(num_chunk_splits)                               \
 105   /* Number of chunks added to freelist as */       \
 106   /* result of a split operation */                 \
 107   x(num_chunks_added_to_freelist_due_to_split)      \
 108   /* Number of chunk in place enlargements */       \
 109   x(num_chunks_enlarged)                            \
 110   /* Number of chunks retired */                    \
 111   x(num_chunks_retired)                             \
 112                                                     \
 113   /* Number of times we did a purge */              \
 114   x(num_purges)                                     \
 115   /* Number of times we did a wholesale uncommit */ \
 116   x(num_wholesale_uncommits)                        \
 117 
 118 
 119 
 120 #define DEFINE_COUNTER(name)          static uintx _##name;
 121 #define DEFINE_ATOMIC_COUNTER(name)   static volatile uintx _##name;
 122 
 123   ALL_MY_COUNTERS(DEFINE_COUNTER, DEFINE_ATOMIC_COUNTER)
 124 
 125 #undef DEFINE_COUNTER
 126 #undef DEFINE_ATOMIC_COUNTER
 127 
 128 public:
 129 
 130 // incrementors
 131 
 132 #define INCREMENTOR(name)           static void inc_##name() { _##name ++; }
 133 #define INCREMENTOR_ATOMIC(name)    static void inc_##name() { Atomic::inc(&_##name); }
 134 
 135   ALL_MY_COUNTERS(INCREMENTOR, INCREMENTOR_ATOMIC)
 136 
 137 #undef INCREMENTOR
 138 #undef INCREMENTOR_ATOMIC
 139 
 140 // getters
 141 
 142 #define GETTER(name)                static uintx name() { return _##name; }
 143 #define GETTER_ATOMIC(name)         GETTER(name)
 144 
 145   ALL_MY_COUNTERS(GETTER, GETTER_ATOMIC)
 146 
 147 #undef GETTER
 148 #undef GETTER_ATOMIC
 149 
 150 
 151   static void print_on(outputStream* st);
 152 
 153 
 154 };
 155 
 156 } // namespace metaspace
 157 
 158 #endif // ASSERT
 159 
 160 #endif // SHARE_MEMORY_METASPACE_INTERNSTAT_HPP
