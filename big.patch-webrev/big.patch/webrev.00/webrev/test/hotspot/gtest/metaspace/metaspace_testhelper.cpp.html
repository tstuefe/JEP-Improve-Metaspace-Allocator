  67 Metachunk* MetaspaceTestHelper::checked_alloc_chunk_0(chunklevel_t preferred_level, chunklevel_t max_level,
  68                                                       size_t min_committed_size, expected_result_t expected_result) {
  69 
  70   Metachunk* c = cm().get_chunk(preferred_level, max_level, min_committed_size);
  71 
  72   if (c != NULL) {
  73 
  74     EXPECT_TRUE(expected_result != result_failure);
  75 
  76     EXPECT_LE(c->level(), max_level);
  77     EXPECT_GE(c->level(), preferred_level);
  78     EXPECT_GE(c->committed_words(), min_committed_size);
  79     EXPECT_EQ(c->committed_words(), c->free_below_committed_words());
  80     EXPECT_EQ(c->used_words(), (size_t)0);
  81     EXPECT_TRUE(c->is_in_use());
  82     EXPECT_FALSE(c->is_free());
  83     EXPECT_FALSE(c->is_dead());
  84     EXPECT_NULL(c->next());
  85     EXPECT_NULL(c->prev());
  86     if (c->level() == HIGHEST_CHUNK_LEVEL) {
  87       EXPECT_TRUE(c->is_leaf_chunk());
  88     } else {
  89       EXPECT_FALSE(c->is_leaf_chunk());
  90     }
  91     if (c->level() == LOWEST_CHUNK_LEVEL) {
  92       EXPECT_TRUE(c->is_root_chunk());
  93     } else {
  94       EXPECT_FALSE(c->is_root_chunk());
  95     }
  96     if (_num_chunks_allocated == 0) { // First chunk? We can make more assumptions
  97       EXPECT_EQ(c->level(), preferred_level);
  98       // Needs lock EXPECT_NULL(c->next_in_vs());
  99       // Needs lock EXPECT_NULL(c->prev_in_vs());
 100       EXPECT_TRUE(c->is_root_chunk() || c->is_leader());
 101     }
 102     _num_chunks_allocated ++;
 103 
 104   } else {
 105 
 106     EXPECT_TRUE(expected_result != result_success);
 107 
 108   }
 109 
 110   DEBUG_ONLY(verify();)
 111 
 112   return c;
 113 }

 115 // Test pattern established when allocating from the chunk with allocate_from_chunk_with_tests().
 116 void MetaspaceTestHelper::test_pattern(Metachunk* c, size_t word_size) {
 117   check_range_for_pattern(c->base(), word_size, (uintx)c);
 118 }

 120 void MetaspaceTestHelper::return_chunk(Metachunk* c) {
 121   test_pattern(c);
 122   DEBUG_ONLY(verify();)
 123   c->set_in_use(); // Forestall assert in cm
 124   _cm.return_chunk(c);
 125 }

 127 MetaWord* MetaspaceTestHelper::allocate_from_chunk(Metachunk* c, size_t word_size) {
 128 
 129   size_t used_before = c->used_words();
 130   size_t free_before = c->free_words();
 131   size_t free_below_committed_before = c->free_below_committed_words();
 132   const MetaWord* top_before = c->top();
 133 
 134   MetaWord* p = c->allocate(word_size);
 135   EXPECT_NOT_NULL(p);
 136   EXPECT_EQ(c->used_words(), used_before + word_size);
 137   EXPECT_EQ(c->free_words(), free_before - word_size);
 138   EXPECT_EQ(c->free_below_committed_words(), free_below_committed_before - word_size);
 139   EXPECT_EQ(c->top(), top_before + word_size);
 140 
 141   // Old content should be preserved
 142   test_pattern(c, used_before);
 143 
 144   // Fill newly allocated range too
 145   fill_range_with_pattern(p, word_size, (uintx)c);
 146 
 147   return p;
 148 }

 150 void MetaspaceTestHelper::commit_chunk_with_test(Metachunk* c, size_t additional_size) {
 151 
 152   size_t used_before = c->used_words();
 153   size_t free_before = c->free_words();
 154   const MetaWord* top_before = c->top();
 155 
 156   c->set_in_use();
 157   bool b = c->ensure_committed_additional(additional_size);
 158   EXPECT_TRUE(b);
 159 
 160   // We should have enough committed size now
 161   EXPECT_GE(c->free_below_committed_words(), additional_size);
 162 
 163   // used, free, top should be unchanged.
 164   EXPECT_EQ(c->used_words(), used_before);
 165   EXPECT_EQ(c->free_words(), free_before);
 166   EXPECT_EQ(c->top(), top_before);
 167 
 168   test_pattern(c, used_before);
 169 
 170 }

 172 void MetaspaceTestHelper::commit_chunk_expect_failure(Metachunk* c, size_t additional_size) {
 173 
 174   size_t used_before = c->used_words();
 175   size_t free_before = c->free_words();
 176   size_t free_below_committed_before = c->free_below_committed_words();
 177   const MetaWord* top_before = c->top();
 178 
 179   c->set_in_use();
 180   bool b = c->ensure_committed_additional(additional_size);
 181   EXPECT_FALSE(b);
 182 
 183   // Nothing should have changed
 184   EXPECT_EQ(c->used_words(), used_before);
 185   EXPECT_EQ(c->free_words(), free_before);
 186   EXPECT_EQ(c->free_below_committed_words(), free_below_committed_before);
 187   EXPECT_EQ(c->top(), top_before);
 188 
 189   test_pattern(c, used_before);
 190 
 191 }

 193 void MetaspaceTestHelper::uncommit_chunk_with_test(Metachunk* c) {
 194   if (c->word_size() >= Settings::commit_granule_words()) {
 195     c->set_free();  // Forestall assert in uncommit
 196     c->reset_used_words();
 197     c->uncommit();
 198 
 199     EXPECT_EQ(c->free_below_committed_words(), (size_t)0);
 200     EXPECT_EQ(c->used_words(), (size_t)0);
 201     EXPECT_EQ(c->free_words(), c->word_size());
 202     EXPECT_EQ(c->top(), c->base());
 203     EXPECT_TRUE(c->is_fully_uncommitted());
 204   }
 205 }
