  30 #include "metaspaceTestsCommon.hpp"
  31 
  32 TEST_VM(metaspace, arena_growth_policy) {
  33 
  34   // These tests may have to be adapted if we change policies
  35 
  36   const ArenaGrowthPolicy* a =
  37       ArenaGrowthPolicy::policy_for_space_type(metaspace::ReflectionMetaspaceType, false);
  38 
  39   EXPECT_EQ(a->get_level_at_step(0), CHUNK_LEVEL_1K);
  40   EXPECT_EQ(a->get_level_at_step(2), CHUNK_LEVEL_1K);
  41   EXPECT_EQ(a->get_level_at_step(10), CHUNK_LEVEL_1K);
  42 
  43   a = ArenaGrowthPolicy::policy_for_space_type(metaspace::ClassMirrorHolderMetaspaceType, false);
  44 
  45   EXPECT_EQ(a->get_level_at_step(0), CHUNK_LEVEL_1K);
  46   EXPECT_EQ(a->get_level_at_step(2), CHUNK_LEVEL_1K);
  47   EXPECT_EQ(a->get_level_at_step(10), CHUNK_LEVEL_1K);
  48 
  49   a = ArenaGrowthPolicy::policy_for_space_type(metaspace::StandardMetaspaceType, false);
  50 
  51   EXPECT_EQ(a->get_level_at_step(0), CHUNK_LEVEL_2K);
  52   EXPECT_EQ(a->get_level_at_step(2), CHUNK_LEVEL_8K);
  53   EXPECT_EQ(a->get_level_at_step(10), CHUNK_LEVEL_64K);
  54 
  55   a = ArenaGrowthPolicy::policy_for_space_type(metaspace::BootMetaspaceType, false);
  56 
  57   EXPECT_EQ(a->get_level_at_step(0), CHUNK_LEVEL_4M);
  58   EXPECT_EQ(a->get_level_at_step(2), CHUNK_LEVEL_1M);
  59   EXPECT_EQ(a->get_level_at_step(10), CHUNK_LEVEL_1M);
  60 
  61 }
