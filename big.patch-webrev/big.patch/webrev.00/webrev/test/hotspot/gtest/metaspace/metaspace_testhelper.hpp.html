  25 #ifndef GTEST_METASPACE_METASPACETESTHELPER_HPP
  26 #define GTEST_METASPACE_METASPACETESTHELPER_HPP
  27 
  28 #include "memory/allocation.hpp"
  29 #include "metaspace/metaspaceTestsCommon.hpp"
  30 
  31 using namespace metaspace::chunklevel;
  32 
  33 class MetaspaceTestHelper : public StackObj {
  34 
  35   ReservedSpace _rs;
  36   CommitLimiter _commit_limiter;
  37   VirtualSpaceList _vs_list;
  38   ChunkManager _cm;
  39   int _num_chunks_allocated;
  40 
  41   enum expected_result_t { result_success = 0, result_failure = 1, result_unknown = 2 };
  42   Metachunk* checked_alloc_chunk_0(chunklevel_t preferred_level, chunklevel_t max_level,
  43                                    size_t min_committed_size, expected_result_t expected_result);
  44 
  45 public:
  46 
  47   // No reserve limit, and a commit limit.
  48   MetaspaceTestHelper(size_t commit_limit);
  49 
  50   // Reserve limit and commit limit.
  51   MetaspaceTestHelper(size_t reserve_limit, size_t commit_limit);
  52 
  53   // Default ctor should cause no limits to fire (within reason)
  54   MetaspaceTestHelper ();
  55 
  56   ~MetaspaceTestHelper();
  57 
  58   const CommitLimiter& commit_limiter() const { return _commit_limiter; }
  59   const VirtualSpaceList& vslist() const { return _vs_list; }
  60   ChunkManager& cm() { return _cm; }
  61 
  62   // Returns reserve- and commit limit we run the test with (in the real world,
  63   // these would be equivalent to CompressedClassSpaceSize resp MaxMetaspaceSize)
  64   size_t reserve_limit() const { return _rs.is_reserved() ? _rs.size() : max_uintx; }
  65   size_t commit_limit() const { return _commit_limiter.cap(); }
  66 
  67   /////
  68 
  69   // Allocate a chunk; do not expect success, but if it succeeds, test the chunk.
  70   Metachunk* alloc_chunk(chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  71     return checked_alloc_chunk_0(preferred_level, max_level, min_committed_size, result_unknown);
  72   }
  73 
  74   // Allocate a chunk; do not expect success, but if it succeeds, test the chunk.
  75   Metachunk* alloc_chunk(chunklevel_t level) {
  76     return alloc_chunk(level, level, word_size_for_level(level));
  77   }
  78 
  79   // Allocate a chunk; it must succeed. Test the chunk.
  80   Metachunk* alloc_chunk_expect_success(chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  81     return checked_alloc_chunk_0(preferred_level, max_level, min_committed_size, result_success);
  82   }
  83 
  84   // Allocate a chunk; it must succeed. Test the chunk.
  85   Metachunk* alloc_chunk_expect_success(chunklevel_t level) {
  86     return alloc_chunk_expect_success(level, level, word_size_for_level(level));
  87   }
  88 
  89   // Allocate a chunk but expect it to fail.
  90   void alloc_chunk_expect_failure(chunklevel_t preferred_level, chunklevel_t max_level, size_t min_committed_size) {
  91     checked_alloc_chunk_0(preferred_level, max_level, min_committed_size, result_failure);
  92   }
  93 
  94   // Allocate a chunk but expect it to fail.
  95   void alloc_chunk_expect_failure(chunklevel_t level) {
  96     return alloc_chunk_expect_failure(level, level, word_size_for_level(level));
  97   }
  98 
  99   /////
 100 
 101   void return_chunk(Metachunk* c);
 102 
 103   /////
 104 
 105   // Allocates from a chunk; also, fills allocated area with test pattern which will be tested with test_pattern().
 106   MetaWord* allocate_from_chunk(Metachunk* c, size_t word_size);
 107 
 108   // Test pattern established when allocating from the chunk with allocate_from_chunk_with_tests().
 109   void test_pattern(Metachunk* c, size_t word_size);
 110   void test_pattern(Metachunk* c) { test_pattern(c, c->used_words()); }
 111 
 112   void commit_chunk_with_test(Metachunk* c, size_t additional_size);
 113   void commit_chunk_expect_failure(Metachunk* c, size_t additional_size);
 114 
 115   void uncommit_chunk_with_test(Metachunk* c);
 116 
 117   DEBUG_ONLY(void verify() const;)
 118 
 119 };
 120 
 121 
 122 #endif // GTEST_METASPACE_METASPACETESTHELPER_HPP
