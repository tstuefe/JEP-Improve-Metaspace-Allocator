  33 TEST_VM(metaspace, metachunklist) {
  34 
  35   MetaspaceTestHelper helper;
  36 
  37   MetachunkList lst;
  38 
  39   Metachunk* chunks[10];
  40   size_t total_size = 0;
  41 
  42   for (int i = 0; i < 10; i ++) {
  43     Metachunk* c = helper.alloc_chunk_expect_success(ChunkLevelRanges::all_chunks().random_value());
  44     chunks[i] = c;
  45     total_size += c->committed_words();
  46 
  47     lst.add(c);
  48     EXPECT_EQ(lst.first(), c);
  49 
  50     Metachunk* c2 = lst.remove_first();
  51     EXPECT_EQ(c, c2);
  52 
  53     EXPECT_EQ(lst.count(), i);
  54     lst.add(c);
  55     EXPECT_EQ(lst.count(), i + 1);
  56     EXPECT_EQ(lst.calc_committed_word_size(), total_size);
  57 
  58   }
  59 
  60   for (int i = 0; i < 10; i ++) {
  61     DEBUG_ONLY(EXPECT_TRUE(lst.contains(chunks[i]));)
  62   }
  63 
  64   for (int i = 0; i < 10; i ++) {
  65     Metachunk* c = lst.remove_first();
  66     DEBUG_ONLY(EXPECT_FALSE(lst.contains(c));)
  67     helper.return_chunk(c);
  68   }
  69 
  70   EXPECT_EQ(lst.count(), 0);
  71   EXPECT_EQ(lst.calc_committed_word_size(), (size_t)0);
  72 
  73 }

  76 TEST_VM(metaspace, freechunklist) {
  77 
  78   MetaspaceTestHelper helper;
  79 
  80   FreeChunkListVector lst;
  81 
  82   MemRangeCounter cnt;
  83   MemRangeCounter committed_cnt;
  84 
  85   // Add random chunks to list and check the counter apis (word_size, commited_word_size, num_chunks)
  86   // Make every other chunk randomly uncommitted, and later we check that committed chunks are sorted in at the front
  87   // of the lists.
  88   for (int i = 0; i < 100; i ++) {
  89     Metachunk* c = helper.alloc_chunk_expect_success(ChunkLevelRanges::all_chunks().random_value());
  90     bool uncommitted_chunk = i % 3;
  91     if (uncommitted_chunk) {
  92       helper.uncommit_chunk_with_test(c);
  93       c->set_in_use();
  94     }
  95 
  96     lst.add(c);
  97 
  98     LOG("->" METACHUNK_FULL_FORMAT, METACHUNK_FULL_FORMAT_ARGS(c));
  99 
 100     cnt.add(c->word_size());
 101     committed_cnt.add(c->committed_words());
 102 
 103     EXPECT_EQ(lst.num_chunks(), (int)cnt.count());
 104     EXPECT_EQ(lst.word_size(), cnt.total_size());
 105     EXPECT_EQ(lst.committed_word_size(), committed_cnt.total_size());
 106   }
 107 
 108   // Drain each list separately
 109   for (chunklevel_t lvl = LOWEST_CHUNK_LEVEL; lvl <= HIGHEST_CHUNK_LEVEL; lvl ++) {
 110     Metachunk* c = lst.remove_first(lvl);
 111     bool found_uncommitted = false;
 112     while (c != NULL) {
 113 
 114       LOG("<-" METACHUNK_FULL_FORMAT, METACHUNK_FULL_FORMAT_ARGS(c));
 115 
 116       // Within a level, no committed chunk should follow an uncommitted chunk:
 117       if (found_uncommitted) {
 118         EXPECT_TRUE(c->is_fully_uncommitted());
 119       } else {
 120         found_uncommitted = c->is_fully_uncommitted();
 121       }
 122 
 123       cnt.sub(c->word_size());
 124       committed_cnt.sub(c->committed_words());
 125 
 126       EXPECT_EQ(lst.num_chunks(), (int)cnt.count());
 127       EXPECT_EQ(lst.word_size(), cnt.total_size());
 128       EXPECT_EQ(lst.committed_word_size(), committed_cnt.total_size());
 129 
 130       helper.return_chunk(c);
 131 
 132       c = lst.remove_first(lvl);
 133     }
 134   }
