  44 static bool use_class_space(bool is_class) {
  45   if (Metaspace::using_class_space() && is_class) {
  46     return true;
  47   }
  48   return false;
  49 }
  50 
  51 static bool use_class_space(Metaspace::MetadataType mdType) {
  52   return use_class_space(is_class(mdType));
  53 }
  54 
  55 ClassLoaderMetaspace::ClassLoaderMetaspace(Mutex* lock, MetaspaceType space_type)
  56   : _lock(lock)
  57   , _space_type(space_type)
  58   , _non_class_space_manager(NULL)
  59   , _class_space_manager(NULL)
  60 {
  61   ChunkManager* const non_class_cm =
  62           ChunkManager::chunkmanager_nonclass();
  63 
  64   // Initialize non-class Arena
  65   _non_class_space_manager = new SpaceManager(
  66       non_class_cm,
  67       ArenaGrowthPolicy::policy_for_space_type(space_type, false),
  68       lock,
  69       RunningCounters::used_nonclass_counter(),
  70       "non-class sm",
  71       is_micro());
  72 
  73   // If needed, initialize class arena
  74   if (Metaspace::using_class_space()) {
  75     ChunkManager* const class_cm =
  76             ChunkManager::chunkmanager_class();
  77     _class_space_manager = new SpaceManager(
  78         class_cm,
  79         ArenaGrowthPolicy::policy_for_space_type(space_type, true),
  80         lock,
  81         RunningCounters::used_class_counter(),
  82         "class sm",
  83         is_micro());
  84   }
  85 
  86 #ifdef ASSERT
  87   InternalStats::inc_num_metaspace_births();
  88   if (_space_type == metaspace::ClassMirrorHolderMetaspaceType) {
  89     InternalStats::inc_num_anon_cld_births();
  90   }
  91 #endif
  92 }
  93 
  94 ClassLoaderMetaspace::~ClassLoaderMetaspace() {
  95   Metaspace::assert_not_frozen();
  96 
  97   delete _non_class_space_manager;
  98   delete _class_space_manager;
  99 
 100 #ifdef ASSERT
 101   InternalStats::inc_num_metaspace_deaths();
 102   if (_space_type == metaspace::ClassMirrorHolderMetaspaceType) {
 103     InternalStats::inc_num_anon_cld_deaths();
 104   }
 105 #endif
 106 
 107 }

 109 // Allocate word_size words from Metaspace.
 110 MetaWord* ClassLoaderMetaspace::allocate(size_t word_size, Metaspace::MetadataType mdType) {
 111   Metaspace::assert_not_frozen();
 112   if (use_class_space(mdType)) {
 113     return class_space_manager()->allocate(word_size);
 114   } else {
 115     return non_class_space_manager()->allocate(word_size);
 116   }
 117 }
 118 
 119 // Attempt to expand the GC threshold to be good for at least another word_size words
 120 // and allocate. Returns NULL if failure. Used during Metaspace GC.
 121 MetaWord* ClassLoaderMetaspace::expand_and_allocate(size_t word_size, Metaspace::MetadataType mdType) {
 122   Metaspace::assert_not_frozen();
 123   size_t delta_bytes = MetaspaceGC::delta_capacity_until_GC(word_size * BytesPerWord);
 124   assert(delta_bytes > 0, "Must be");
 125 
 126   size_t before = 0;
 127   size_t after = 0;
 128   bool can_retry = true;
 129   MetaWord* res;
 130   bool incremented;
 131 
 132   // Each thread increments the HWM at most once. Even if the thread fails to increment
 133   // the HWM, an allocation is still attempted. This is because another thread must then
 134   // have incremented the HWM and therefore the allocation might still succeed.
 135   do {
 136     incremented = MetaspaceGC::inc_capacity_until_GC(delta_bytes, &after, &before, &can_retry);
 137     res = allocate(word_size, mdType);
 138   } while (!incremented && res == NULL && can_retry);
 139 
 140   if (incremented) {
 141     Metaspace::tracer()->report_gc_threshold(before, after,
 142                                   MetaspaceGCThresholdUpdater::ExpandAndAllocate);
 143     log_trace(gc, metaspace)("Increase capacity to GC from " SIZE_FORMAT " to " SIZE_FORMAT, before, after);
 144   }
 145 
 146   return res;
 147 }

 149 // Prematurely returns a metaspace allocation to the _block_freelists
 150 // because it is not needed anymore.
 151 void ClassLoaderMetaspace::deallocate(MetaWord* ptr, size_t word_size, bool is_class) {
 152 
 153   Metaspace::assert_not_frozen();
 154 
 155   if (use_class_space(is_class)) {
 156     class_space_manager()->deallocate(ptr, word_size);
 157   } else {
 158     non_class_space_manager()->deallocate(ptr, word_size);
 159   }
 160 
 161   DEBUG_ONLY(InternalStats::inc_num_deallocs();)
 162 
 163 }

 165 // Update statistics. This walks all in-use chunks.
 166 void ClassLoaderMetaspace::add_to_statistics(clms_stats_t* out) const {
 167   if (non_class_space_manager() != NULL) {
 168     non_class_space_manager()->add_to_statistics(&out->sm_stats_nonclass);
 169   }
 170   if (class_space_manager() != NULL) {
 171     class_space_manager()->add_to_statistics(&out->sm_stats_class);
 172   }
 173 }
 174 
 175 #ifdef ASSERT
 176 void ClassLoaderMetaspace::verify() const {
 177   check_valid_spacetype(_space_type);
 178   if (non_class_space_manager() != NULL) {
 179     non_class_space_manager()->verify(false);
 180   }
 181   if (class_space_manager() != NULL) {
 182     class_space_manager()->verify(false);
 183   }
 184 }
 185 #endif // ASSERT

 188 // This only exists for JFR and jcmd VM.classloader_stats. We may want to
 189 //  change this. Capacity as a stat is of questionable use since it may
 190 //  contain committed and uncommitted areas. For now we do this to maintain
 191 //  backward compatibility with JFR.
 192 void ClassLoaderMetaspace::calculate_jfr_stats(size_t* p_used_bytes, size_t* p_capacity_bytes) const {
 193 
 194   // Implement this using the standard statistics objects.
 195   clms_stats_t stat;
 196   add_to_statistics(&stat);
 197   in_use_chunk_stats_t in_use_chunks_totals = stat.totals().totals();
 198   *p_used_bytes = in_use_chunks_totals.used_words * BytesPerWord;
 199   *p_capacity_bytes = in_use_chunks_totals.word_size * BytesPerWord;
 200 
 201 }
