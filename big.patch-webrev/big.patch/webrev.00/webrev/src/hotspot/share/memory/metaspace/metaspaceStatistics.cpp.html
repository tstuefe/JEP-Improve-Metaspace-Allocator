  40 // Returns total word size of all chunks in this manager.
  41 void cm_stats_t::add(const cm_stats_t& other) {
  42   for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
  43     num_chunks[l] += other.num_chunks[l];
  44     committed_word_size[l] += other.committed_word_size[l];
  45   }
  46 }
  47 
  48 // Returns total word size of all chunks in this manager.
  49 size_t cm_stats_t::total_word_size() const {
  50   size_t s = 0;
  51   for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
  52     s += num_chunks[l] * chunklevel::word_size_for_level(l);
  53   }
  54   return s;
  55 }
  56 
  57 // Returns total committed word size of all chunks in this manager.
  58 size_t cm_stats_t::total_committed_word_size() const {
  59   size_t s = 0;
  60   for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
  61     s += committed_word_size[l];
  62   }
  63   return s;
  64 }
  65 
  66 
  67 void cm_stats_t::print_on(outputStream* st, size_t scale) const {
  68   // Note: used as part of MetaspaceReport so formatting matters.
  69   size_t total_size = 0;
  70   size_t total_committed_size = 0;
  71   for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
  72     st->cr();
  73     chunklevel::print_chunk_size(st, l);
  74     st->print(": ");
  75     if (num_chunks[l] > 0) {
  76       const size_t word_size = num_chunks[l] * chunklevel::word_size_for_level(l);
  77 
  78       st->print("%4d, capacity=", num_chunks[l]);
  79       print_scaled_words(st, word_size, scale);
  80 
  81       st->print(", committed=");
  82       print_scaled_words_and_percentage(st, committed_word_size[l], word_size, scale);
  83 
  84       total_size += word_size;
  85       total_committed_size += committed_word_size[l];
  86     } else {
  87       st->print("(none)");
  88     }
  89   }
  90   st->cr();
  91   st->print("Total word size: ");
  92   print_scaled_words(st, total_size, scale);
  93   st->print(", committed: ");
  94   print_scaled_words_and_percentage(st, total_committed_size, total_size, scale);
  95   st->cr();
  96 }
  97 
  98 #ifdef ASSERT
  99 void cm_stats_t::verify() const {
 100   assert(total_committed_word_size() <= total_word_size(),
 101          "Sanity");
 102 }
 103 #endif

 107 void in_use_chunk_stats_t::print_on(outputStream* st, size_t scale) const {
 108   int col = st->position();
 109   st->print("%4d chunk%s, ", num, num != 1 ? "s" : "");
 110   if (num > 0) {
 111     col += 14; st->fill_to(col);
 112 
 113     print_scaled_words(st, word_size, scale, 5);
 114     st->print(" capacity,");
 115 
 116     col += 20; st->fill_to(col);
 117     print_scaled_words_and_percentage(st, committed_words, word_size, scale, 5);
 118     st->print(" committed, ");
 119 
 120     col += 18; st->fill_to(col);
 121     print_scaled_words_and_percentage(st, used_words, word_size, scale, 5);
 122     st->print(" used, ");
 123 
 124     col += 20; st->fill_to(col);
 125     print_scaled_words_and_percentage(st, free_words, word_size, scale, 5);
 126     st->print(" free, ");
 127 
 128     col += 20; st->fill_to(col);
 129     print_scaled_words_and_percentage(st, waste_words, word_size, scale, 5);
 130     st->print(" waste ");
 131 
 132   }
 133 }
 134 
 135 #ifdef ASSERT
 136 void in_use_chunk_stats_t::verify() const {
 137   assert(word_size >= committed_words &&
 138       committed_words == used_words + free_words + waste_words,
 139          "Sanity: cap " SIZE_FORMAT ", committed " SIZE_FORMAT ", used " SIZE_FORMAT ", free " SIZE_FORMAT ", waste " SIZE_FORMAT ".",
 140          word_size, committed_words, used_words, free_words, waste_words);
 141 }
 142 #endif

 144 // SpaceManagerStatistics methods
 145 
 146 void sm_stats_t::add(const sm_stats_t& other) {
 147   for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
 148     stats[l].add(other.stats[l]);
 149   }
 150   free_blocks_num += other.free_blocks_num;
 151   free_blocks_word_size += other.free_blocks_word_size;
 152 }
 153 
 154 
 155 // Returns total chunk statistics over all chunk types.
 156 in_use_chunk_stats_t sm_stats_t::totals() const {
 157   in_use_chunk_stats_t out;
 158   for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
 159     out.add(stats[l]);
 160   }
 161   return out;
 162 }
 163 
 164 void sm_stats_t::print_on(outputStream* st, size_t scale,  bool detailed) const {
 165   streamIndentor sti(st);
 166   if (detailed) {
 167     st->cr_indent();
 168     st->print("Usage by chunk level:");
 169     {
 170       streamIndentor sti2(st);
 171       for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
 172         st->cr_indent();
 173         chunklevel::print_chunk_size(st, l);
 174         st->print(" chunks: ");
 175         if (stats[l].num == 0) {
 176           st->print(" (none)");
 177         } else {
 178           stats[l].print_on(st, scale);
 179         }
 180       }
 181 
 182       st->cr_indent();
 183       st->print("%15s: ", "-total-");
 184       totals().print_on(st, scale);
 185     }
 186     if (free_blocks_num > 0) {
 187       st->cr_indent();
 188       st->print("deallocated: " UINTX_FORMAT " blocks with ", free_blocks_num);
 189       print_scaled_words(st, free_blocks_word_size, scale);
 190     }
 191   } else {
 192     totals().print_on(st, scale);
 193     st->print(", ");
 194     st->print("deallocated: " UINTX_FORMAT " blocks with ", free_blocks_num);
 195     print_scaled_words(st, free_blocks_word_size, scale);
 196   }
 197 }

 199 #ifdef ASSERT
 200 
 201 void sm_stats_t::verify() const {
 202   size_t total_used = 0;
 203   for (chunklevel_t l = chunklevel::LOWEST_CHUNK_LEVEL; l <= chunklevel::HIGHEST_CHUNK_LEVEL; l ++) {
 204     stats[l].verify();
 205     total_used += stats[l].used_words;
 206   }
 207   // Deallocated allocations still count as used
 208   assert(total_used >= free_blocks_word_size,
 209          "Sanity");
 210 }
 211 #endif

 215 // Returns total space manager statistics for both class and non-class metaspace
 216 sm_stats_t clms_stats_t::totals() const {
 217   sm_stats_t out;
 218   out.add(sm_stats_nonclass);
 219   out.add(sm_stats_class);
 220   return out;
 221 }
 222 
 223 void clms_stats_t::print_on(outputStream* st, size_t scale, bool detailed) const {
 224   streamIndentor sti(st);
 225   st->cr_indent();
 226   if (Metaspace::using_class_space()) {
 227     st->print("Non-Class: ");
 228   }
 229   sm_stats_nonclass.print_on(st, scale, detailed);
 230   if (detailed) {
 231     st->cr();
 232   }
 233   if (Metaspace::using_class_space()) {
 234     st->cr_indent();
 235     st->print("    Class: ");
 236     sm_stats_class.print_on(st, scale, detailed);
 237     if (detailed) {
 238       st->cr();
 239     }
 240     st->cr_indent();
 241     st->print("     Both: ");
 242     totals().print_on(st, scale, detailed);
 243     if (detailed) {
 244       st->cr();
 245     }
 246   }
 247   st->cr();
 248 }
