  37 // We seem to be counting a lot of things which makes it worthwhile to
  38 // make helper classes for all that boilerplate coding.
  39 
  40 // AbstractCounter counts something and asserts overflow and underflow.
  41 template <class T>
  42 class AbstractCounter {
  43 
  44   T _c;
  45 
  46   // Only allow unsigned values for now
  47   STATIC_ASSERT(IsSigned<T>::value == false);
  48 
  49 public:
  50 
  51   AbstractCounter() : _c(0) {}
  52 
  53   T get() const           { return _c; }
  54 
  55   void increment() { increment_by(1); }
  56   void decrement() { decrement_by(1); }
  57 
  58   void increment_by(T v) {
  59     assert(_c + v >= _c,
  60         "overflow (" UINT64_FORMAT "+" UINT64_FORMAT ")",
  61         (uint64_t)_c, (uint64_t)v);
  62     _c += v;
  63   }
  64 
  65   void decrement_by(T v) {
  66     assert(_c - v <= _c,
  67         "underflow (" UINT64_FORMAT "-" UINT64_FORMAT ")",
  68         (uint64_t)_c, (uint64_t)v);
  69     _c -= v;
  70   }
  71 
  72   void reset()                { _c = 0; }
  73 
  74 #ifdef ASSERT
  75   void check(T expected) const {
  76     assert(_c == expected, "Counter mismatch: %d, expected: %d.",
  77            (int)_c, (int)expected);
  78     }
  79 #endif
  80 
  81 };
  82 
  83 // Atomic variant of AbstractCounter.
  84 template <class T>
  85 class AbstractAtomicCounter {
  86 
  87   volatile T _c;
  88 
  89   // Only allow unsigned values for now
  90   STATIC_ASSERT(IsSigned<T>::value == false);
  91 
  92 public:
  93 
  94   AbstractAtomicCounter() : _c(0) {}
  95 
  96   T get() const               { return _c; }
  97 
  98   void increment() {
  99     assert(_c + 1 != 0,
 100         "overflow (" UINT64_FORMAT "+1)", (uint64_t)_c);
 101     Atomic::inc(&_c);
 102   }
 103 
 104   void decrement() {
 105     assert(_c >= 1,
 106         "underflow (" UINT64_FORMAT "-1)", (uint64_t)_c);
 107     Atomic::dec(&_c);
 108   }
 109 
 110   void increment_by(T v) {
 111     T v1 = _c;
 112     T v2 = Atomic::add(&_c, v);
 113     assert(v2 > v1,
 114            "overflow (" UINT64_FORMAT "+" UINT64_FORMAT ")",
 115            (uint64_t)v1, (uint64_t)v);
 116   }
 117 
 118   void decrement_by(T v) {
 119     assert(_c >= v,
 120            "underflow (" UINT64_FORMAT "-" UINT64_FORMAT ")",
 121            (uint64_t)_c, (uint64_t)v);
 122     Atomic::sub(&_c, v);
 123   }
 124 
 125 #ifdef ASSERT
 126   void check(T expected) const {
 127     assert(_c == expected, "Counter mismatch: %d, expected: %d.",
 128            (int)_c, (int)expected);
 129     }
 130 #endif
 131 
 132 };
 133 
 134 typedef AbstractCounter<size_t>   SizeCounter;
 135 typedef AbstractCounter<unsigned> IntCounter;
 136 
 137 typedef AbstractAtomicCounter<size_t> SizeAtomicCounter;

 140 // We often count memory ranges (blocks, chunks etc).
 141 // Make a helper class for that.
 142 template <class T_num, class T_size>
 143 class AbstractMemoryRangeCounter {
 144 
 145   AbstractCounter<T_num>  _count;
 146   AbstractCounter<T_size> _total_size;
 147 
 148 public:
 149 
 150   void add(T_size s) {
 151     if(s > 0) {
 152       _count.increment();
 153       _total_size.increment_by(s);
 154     }
 155   }
 156 
 157   void sub(T_size s) {
 158     if(s > 0) {
 159       _count.decrement();
 160       _total_size.decrement_by(s);
 161     }
 162   }
 163 
 164   T_num count() const       { return _count.get(); }
 165   T_size total_size() const { return _total_size.get(); }
 166 
 167 
 168 #ifdef ASSERT
 169   void check(T_num expected_count, T_size expected_size) const {
 170     _count.check(expected_count);
 171     _total_size.check(expected_size);
 172   }
 173   void check(const AbstractMemoryRangeCounter<T_num, T_size>& other) const {
 174     check(other.count(), other.total_size());
 175   }
 176 #endif
 177 
 178 };
 179 
 180 typedef AbstractMemoryRangeCounter<unsigned, size_t> MemRangeCounter;
