  26 #ifndef SHARE_MEMORY_METASPACE_CHUNKHEADERPOOL_HPP
  27 #define SHARE_MEMORY_METASPACE_CHUNKHEADERPOOL_HPP
  28 
  29 #include "memory/metaspace/internStat.hpp"
  30 #include "memory/metaspace/metachunk.hpp"
  31 #include "memory/metaspace/metachunkList.hpp"
  32 #include "utilities/debug.hpp"
  33 #include "utilities/globalDefinitions.hpp"

  35 namespace metaspace {
  36 
  37 // Chunk headers are kept separate from the chunks themselves; in order to
  38 //  speed up allocation, reduce waste and increase locality when walking chains
  39 //  of linked headers, they are kept in a pool.
  40 //
  41 // The ChunkHeaderPool is a growable array of chunk headers. Unused chunk headers
  42 //  are kept in a free list.
  43 
  44 class ChunkHeaderPool {
  45 
  46   static const int slab_capacity = 128;
  47 
  48   struct slab_t : public CHeapObj<mtMetaspace> {
  49     slab_t* next;
  50     int top;
  51     Metachunk elems [slab_capacity];
  52     slab_t() : next(NULL), top(0) {
  53       for (int i = 0; i < slab_capacity; i++) {
  54         elems[i].clear();
  55       }
  56     }
  57   };
  58 
  59   IntCounter _num_slabs;
  60   slab_t* _first_slab;
  61   slab_t* _current_slab;
  62 
  63   IntCounter _num_handed_out;
  64 
  65   MetachunkList _freelist;
  66 
  67   const bool _delete_on_destruction;
  68 
  69   void allocate_new_slab();
  70 
  71   static ChunkHeaderPool _chunkHeaderPool;
  72 
  73 
  74 public:
  75 
  76   ChunkHeaderPool(bool delete_on_destruction);
  77 
  78   ~ChunkHeaderPool();
  79 
  80   // Allocates a Metachunk structure. The structure is uninitialized.
  81   Metachunk* allocate_chunk_header() {
  82 
  83     Metachunk* c = NULL;
  84 
  85     DEBUG_ONLY(verify(false));
  86 
  87     c = _freelist.remove_first();
  88     assert(c == NULL || c->is_dead(), "Not a freelist chunk header?");
  89 
  90     if (c == NULL) {
  91 
  92       if (_current_slab == NULL ||
  93           _current_slab->top == slab_capacity) {
  94         allocate_new_slab();
  95         assert(_current_slab->top < slab_capacity, "Sanity");
  96       }
  97 
  98       c = _current_slab->elems + _current_slab->top;
  99       _current_slab->top ++;
 100 
 101     }
 102 
 103     _num_handed_out.increment();
 104 
 105     // By contract, the returned structure is uninitialized.
 106     // Zap to make this clear.
 107     DEBUG_ONLY(c->zap_header(0xBB);)
 108 
 109     return c;
 110 
 111   }
 112 
 113   void return_chunk_header(Metachunk* c) {
 114     // We only ever should return free chunks, since returning chunks
 115     // happens only on merging and merging only works with free chunks.
 116     assert(c != NULL && c->is_free(), "Sanity");
 117 #ifdef ASSERT
 118     // In debug, fill dead header with pattern.
 119     c->zap_header(0xCC);
 120     c->set_next(NULL);
 121     c->set_prev(NULL);
 122 #endif
 123     c->set_dead();
 124     _freelist.add(c);
 125     _num_handed_out.decrement();
 126 
 127   }
 128 
 129   // Returns number of allocated elements.
 130   int used() const                   { return _num_handed_out.get(); }
 131 
 132   // Returns number of elements in free list.
 133   int freelist_size() const          { return _freelist.count(); }
 134 
 135   // Returns size of memory used.
 136   size_t memory_footprint_words() const;
 137 
 138   DEBUG_ONLY(void verify(bool slow) const;)
 139 
 140   // Returns reference to the one global chunk header pool.
 141   static ChunkHeaderPool& pool() { return _chunkHeaderPool; }
 142 
 143 };
