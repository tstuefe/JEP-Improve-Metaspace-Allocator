  35 // A growth policy which is fixed. Last growth step is endlessly repeated
  36 // (so from then on, the growth is linear)
  37 
  38 class ConstantArenaGrowthPolicy : public ArenaGrowthPolicy {
  39 private:
  40 
  41   // An array of sizes defining the arena growth per step. The last size
  42   // is endlessly repeated, so from then on the growth is linear.
  43   const chunklevel_t* const _steps;
  44   const int _num;
  45 
  46 public:
  47 
  48   ConstantArenaGrowthPolicy(const chunklevel_t* steps, int num)
  49     : _steps(steps),
  50       _num(num)
  51   {
  52     assert(_num > 0 && _steps != NULL, "Sanity");
  53   }
  54 
  55   chunklevel_t get_level_at_step(int step) const {
  56     assert(step >= 0, "Sanity");
  57     if (step >= _num) {
  58       return _steps[_num - 1];
  59     }
  60     return _steps[step];
  61   }
  62 
  63 };
  64 
  65 // hard-coded policies sequences for standard space types
  66 
  67 static const chunklevel_t g_sequ_standard_non_class[] = {
  68     chunklevel::CHUNK_LEVEL_2K,
  69     chunklevel::CHUNK_LEVEL_4K,
  70     chunklevel::CHUNK_LEVEL_8K,
  71     chunklevel::CHUNK_LEVEL_64K
  72     // .. repeat last
  73 };
  74 
  75 static const chunklevel_t g_sequ_standard_class[] = {
  76     chunklevel::CHUNK_LEVEL_1K,
  77     chunklevel::CHUNK_LEVEL_2K,
  78     chunklevel::CHUNK_LEVEL_4K,
  79     chunklevel::CHUNK_LEVEL_16K
  80     // .. repeat last
  81 };
  82 
  83 // Policies for micro loaders which are expected to load only one class:
  84 static const chunklevel_t g_sequ_micro[] = {
  85     chunklevel::CHUNK_LEVEL_1K
  86     // .. repeat last
  87 };
  88 
  89 // Boot class loader: we allow it to grow in large steps, and give it
  90 // a large initial step to start. Note that for growth sizes beyond
  91 // commit granule size the costs diminish since the chunks are committed
  92 // on demand only.
  93 static const chunklevel_t g_sequ_boot_non_class[] = {
  94     chunklevel::CHUNK_LEVEL_4M,
  95     chunklevel::CHUNK_LEVEL_1M
  96     // .. repeat last
  97 };
  98 
  99 static const chunklevel_t g_sequ_boot_class[] = {
 100     chunklevel::CHUNK_LEVEL_1M,
 101     chunklevel::CHUNK_LEVEL_256K
 102     // .. repeat last
 103 };
 104 
 105 #define DEFINE_POLICY(what) \
 106   static ConstantArenaGrowthPolicy g_policy_##what (g_sequ_##what, sizeof(g_sequ_##what)/sizeof(chunklevel_t));
 107 
 108 DEFINE_POLICY(standard_non_class)
 109 DEFINE_POLICY(standard_class)
 110 DEFINE_POLICY(micro)
 111 DEFINE_POLICY(boot_non_class)
 112 DEFINE_POLICY(boot_class)
 113 
 114 const ArenaGrowthPolicy* ArenaGrowthPolicy::policy_for_space_type(MetaspaceType space_type, bool is_class) {
 115   switch(space_type) {
 116     case StandardMetaspaceType:          return is_class ? &g_policy_standard_class : &g_policy_standard_non_class;
 117     case BootMetaspaceType:              return is_class ? &g_policy_boot_class : &g_policy_boot_non_class;
 118     case ReflectionMetaspaceType:
 119     case ClassMirrorHolderMetaspaceType: return &g_policy_micro;
 120     default: ShouldNotReachHere();
 121   }
 122 
 123   return NULL;
 124 
 125 }
