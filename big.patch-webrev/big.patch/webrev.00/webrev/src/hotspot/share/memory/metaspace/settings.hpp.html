  26 #ifndef SHARE_MEMORY_METASPACE_CONSTANTS_HPP
  27 #define SHARE_MEMORY_METASPACE_CONSTANTS_HPP
  28 
  29 #include "memory/allocation.hpp"
  30 #include "memory/metaspace/chunkLevel.hpp"
  31 #include "utilities/globalDefinitions.hpp"
  32 
  33 namespace metaspace {
  34 
  35 class Settings : public AllStatic {
  36 
  37   // Granularity, in bytes, metaspace is committed with.
  38   static size_t _commit_granule_bytes;
  39 
  40   // Granularity, in words, metaspace is committed with.
  41   static size_t _commit_granule_words;
  42 
  43   // Whether or not commit new-born root chunks thru after creation.
  44   static bool _newborn_root_chunks_are_fully_committed;
  45 
  46   // When a chunk is handed out by the ChunkManager to a class loader, how much
  47   // of a chunk should be committed up-front?
  48   // Size in words. Will be rounded up to the nearest multiple of commit_granule_words.
  49   // (Note: 0 is possible but inefficient, since it will cause the ClassLoaderMetaspace
  50   //        to commit the first granule right away anyway, so nothing is saved.
  51   //        chunklevel::MAX_CHUNK_WORD_SIZE pretty much means every chunk is committed thru
  52   //        from the start.
  53   static size_t _committed_words_on_fresh_chunks;
  54 
  55   // The default size of a non-class VirtualSpaceNode (unless created differently).
  56   // Must be a multiple of the root chunk size.
  57   static const size_t _virtual_space_node_default_word_size = chunklevel::MAX_CHUNK_WORD_SIZE * 2; // lets go with 8mb virt size. Seems a good compromise betw. virt and mapping fragmentation.
  58 
  59   static const size_t _allocation_from_dictionary_limit = 4 * K;
  60 
  61   // When allocating from a chunk, if the remaining area in the chunk is too small to hold
  62   // the requested size, we attempt to double the chunk size in place...
  63   static const bool _enlarge_chunks_in_place = true;
  64 
  65   // .. but we do only do this for chunks below a given size to prevent unnecessary memory blowup.
  66   static const size_t _enlarge_chunks_in_place_max_word_size = 256 * K;
  67 
  68   // If true, chunks are uncommitted after gc (when metaspace is purged).
  69   static bool _uncommit_on_return;
  70 
  71   // If true, vsnodes which only contain free chunks will be deleted (purged) as part of a gc.
  72   static bool _delete_nodes_on_purge;
  73 
  74   // If _uncommit_on_return is true:
  75   // Minimum word size a chunk has to have after returning and merging with neighboring free chunks
  76   // to be candidate for uncommitting. Must be a multiple of and not smaller than commit granularity.
  77   static size_t _uncommit_on_return_min_word_size;
  78 
  79   // If true, chunks are uncommitted after gc (when metaspace is purged).
  80   static bool _uncommit_on_purge;
  81 
  82   // If _uncommit_on_purge is true:
  83   // Minimum word size of an area to be candidate for uncommitting.
  84   // Must be a multiple of and not smaller than commit granularity.
  85   static size_t _uncommit_on_purge_min_word_size;
  86 
  87   // If true, metablock allocations are guarded and periodically checked.
  88   DEBUG_ONLY(static bool _use_allocation_guard;)
  89 
  90   // If true, we handle deallocated blocks (default).
  91   DEBUG_ONLY(static bool _handle_deallocations;)
  92 
  93 public:
  94 
  95   static size_t commit_granule_bytes()                        { return _commit_granule_bytes; }
  96   static size_t commit_granule_words()                        { return _commit_granule_words; }
  97   static bool newborn_root_chunks_are_fully_committed()       { return _newborn_root_chunks_are_fully_committed; }
  98   static size_t committed_words_on_fresh_chunks()             { return _committed_words_on_fresh_chunks; }
  99   static size_t virtual_space_node_default_word_size()        { return _virtual_space_node_default_word_size; }
 100   static size_t allocation_from_dictionary_limit()            { return _allocation_from_dictionary_limit; }
 101   static bool enlarge_chunks_in_place()                       { return _enlarge_chunks_in_place; }
 102   static size_t enlarge_chunks_in_place_max_word_size()       { return _enlarge_chunks_in_place_max_word_size; }
 103   static bool uncommit_on_return()                            { return _uncommit_on_return; }
 104   static size_t uncommit_on_return_min_word_size()            { return _uncommit_on_return_min_word_size; }
 105   static bool delete_nodes_on_purge()                         { return _delete_nodes_on_purge; }
 106   static bool uncommit_on_purge()                             { return _uncommit_on_purge; }
 107   static size_t uncommit_on_purge_min_word_size()             { return _uncommit_on_purge_min_word_size; }
 108 
 109   static bool use_allocation_guard()                          { return DEBUG_ONLY(_use_allocation_guard) NOT_DEBUG(false); }
 110   static bool handle_deallocations()                          { return DEBUG_ONLY(_handle_deallocations) NOT_DEBUG(true); }
 111 
 112   static void ergo_initialize();
 113 
 114   static void print_on(outputStream* st);
 115 
 116 };
 117 
 118 } // namespace metaspace
 119 
 120 #endif // SHARE_MEMORY_METASPACE_BLOCKFREELIST_HPP
