  35 // Metaspace
  36 //
  37 // Metaspaces are Arenas for the VM's metadata.
  38 // They are allocated one per class loader object, and one for the null
  39 // bootstrap class loader
  40 //
  41 //    block X ---+       +-------------------+
  42 //               |       |  Virtualspace     |
  43 //               |       |                   |
  44 //               |       |                   |
  45 //               |       |-------------------|
  46 //               |       || Chunk            |
  47 //               |       ||                  |
  48 //               |       ||----------        |
  49 //               +------>||| block 0 |       |
  50 //                       ||----------        |
  51 //                       ||| block 1 |       |
  52 //                       ||----------        |
  53 //                       ||                  |
  54 //                       |-------------------|
  55 //                       |                   |
  56 //                       |                   |
  57 //                       +-------------------+
  58 //

  70 ////////////////// Metaspace ///////////////////////
  71 
  72 // Metaspaces each have a  SpaceManager and allocations
  73 // are done by the SpaceManager.  Allocations are done
  74 // out of the current Metachunk.  When the current Metachunk
  75 // is exhausted, the SpaceManager gets a new one from
  76 // the current VirtualSpace.  When the VirtualSpace is exhausted
  77 // the SpaceManager gets a new one.  The SpaceManager
  78 // also manages freelists of available Chunks.
  79 //
  80 // Currently the space manager maintains the list of
  81 // virtual spaces and the list of chunks in use.  Its
  82 // allocate() method returns a block for use as a
  83 // quantum of metadata.
  84

 186 // Metaspace are deallocated when their class loader are GC'ed.
 187 // This class implements a policy for inducing GC's to recover
 188 // Metaspaces.





























 189 
 190 class MetaspaceGCThresholdUpdater : public AllStatic {
 191  public:
 192   enum Type {
 193     ComputeNewSize,
 194     ExpandAndAllocate,
 195     Last

























































































































 196   };
 197 
 198   static const char* to_string(MetaspaceGCThresholdUpdater::Type updater) {
 199     switch (updater) {
 200       case ComputeNewSize:
 201         return "compute_new_size";
 202       case ExpandAndAllocate:
 203         return "expand_and_allocate";
 204       default:
 205         assert(false, "Got bad updater: %d", (int) updater);
 206         return NULL;
 207     };
 208   }













 209 };

 211 class MetaspaceGC : public AllStatic {




 212 
 213   // The current high-water-mark for inducing a GC.
 214   // When committed memory of all metaspaces reaches this value,
 215   // a GC is induced and the value is increased. Size is in bytes.
 216   static volatile size_t _capacity_until_GC;
 217   static uint _shrink_factor;
 218 
 219   static size_t shrink_factor() { return _shrink_factor; }
 220   void set_shrink_factor(uint v) { _shrink_factor = v; }
 221 
 222  public:
 223 
 224   static void initialize();
 225   static void post_initialize();
 226 
 227   static size_t capacity_until_GC();
 228   static bool inc_capacity_until_GC(size_t v,
 229                                     size_t* new_cap_until_GC = NULL,
 230                                     size_t* old_cap_until_GC = NULL,
 231                                     bool* can_retry = NULL);
 232   static size_t dec_capacity_until_GC(size_t v);
 233 
 234   // The amount to increase the high-water-mark (_capacity_until_GC)
 235   static size_t delta_capacity_until_GC(size_t bytes);
 236 
 237   // Tells if we have can expand metaspace without hitting set limits.
 238   static bool can_expand(size_t words, bool is_class);
 239 
 240   // Returns amount that we can expand without hitting a GC,
 241   // measured in words.
 242   static size_t allowed_expansion();
 243 
 244   // Calculate the new high-water mark at which to induce
 245   // a GC.
 246   static void compute_new_size();
 247 };
