  25 #ifndef SHARE_MEMORY_METASPACE_INTERNSTAT_HPP
  26 #define SHARE_MEMORY_METASPACE_INTERNSTAT_HPP

  28 #ifdef ASSERT
  29 
  30 
  31 #include "memory/allocation.hpp"
  32 #include "runtime/atomic.hpp"
  33 #include "utilities/globalDefinitions.hpp"
  34 
  35 class outputStream;
  36 
  37 namespace metaspace {
  38 
  39 
  40 // A number of counters for internal metaspace statistics; Only active in debug.
  41 class InternalStats : public AllStatic {
  42 
  43   // Note: all counters which are modified on the classloader local allocation path
  44   //   (not under ExpandLock protection) have to be atomic.
  45 
  46 #define ALL_MY_COUNTERS(x, x_atomic)                \
  47                                                     \
  48   /* Number of allocations. */                      \
  49   x_atomic(num_allocs)                              \
  50                                                     \
  51   /* Number of deallocations (external) */          \
  52   x_atomic(num_deallocs)                            \
  53   /* Number of times an allocation was satisfied */ \
  54   /*  from deallocated blocks. */                   \
  55   x_atomic(num_allocs_from_deallocated_blocks)      \
  56                                                     \
  57   /* Number of times an allocation failed */        \
  58   /*  because the chunk was too small. */           \
  59         x_atomic(num_allocs_failed_chunk_too_small)       \
  60                                                     \
  61   /* Number of times an allocation failed */        \
  62   /*  because we hit a limit. */                    \
  63   x_atomic(num_allocs_failed_limit)                 \
  64                                                     \
  65   /* Number of times a ClassLoaderMetaspace was */  \
  66   /*  born... */                                    \
  67   x(num_metaspace_births)                           \
  68   /* ... and died. */                               \
  69   x(num_metaspace_deaths)                           \
  70                                                     \
  71         /* Number of times a ClassLoaderMetaspace was */  \
  72         /*  born for a hidden or anonymous class... */    \
  73         x(num_anon_cld_births)                            \
  74         /* ... and died. */                               \
  75   x(num_anon_cld_deaths)                            \
  76                                                     \
  77   /* Number of times VirtualSpaceNode were */       \
  78   /*  created...  */                                \
  79   x(num_vsnodes_created)                            \
  80   /* ... and purged. */                             \
  81   x(num_vsnodes_destroyed)                          \
  82                                                     \
  83   /* Number of times we committed space. */         \
  84   x(num_space_committed)                            \
  85   /* Number of times we uncommitted space. */       \
  86   x(num_space_uncommitted)                          \
  87                                                     \
  88   /* Number of times a chunk was returned to the */ \
  89   /*  freelist (external only). */                  \
  90   x(num_chunks_returned_to_freelist)                \
  91   /* Number of times a chunk was taken from */      \
  92   /*  freelist (external only) */                   \
  93   x(num_chunks_taken_from_freelist)                 \
  94                                                     \
  95   /* Number of successful chunk merges */           \
  96   x(num_chunk_merges)                               \
  97   /* Number of chunks removed from freelist as */   \
  98   /* result of a merge operation */                 \
  99   x(num_chunks_removed_from_freelist_due_to_merge)  \
 100                                                     \
 101   /* Number of chunk splits */                      \
 102   x(num_chunk_splits)                               \
 103   /* Number of chunks added to freelist as */       \
 104   /* result of a split operation */                 \
 105   x(num_chunks_added_to_freelist_due_to_split)      \
 106   /* Number of chunk in place enlargements */       \
 107   x(num_chunk_enlarged)                             \
 108   /* Number of chunks retired */                    \
 109   x(num_chunks_retired)                             \
 110                                                     \
 111   /* Number of times we did a purge */              \
 112   x(num_purges)                                     \
 113   /* Number of times we did a wholesale uncommit */ \
 114   x(num_wholesale_uncommits)                        \
 115 
 116 
 117 
 118 #define DEFINE_COUNTER(name)          static uintx _##name;
 119 #define DEFINE_ATOMIC_COUNTER(name)   static volatile uintx _##name;
 120 
 121   ALL_MY_COUNTERS(DEFINE_COUNTER, DEFINE_ATOMIC_COUNTER)
 122 
 123 #undef DEFINE_COUNTER
 124 #undef DEFINE_ATOMIC_COUNTER
 125 
 126 public:
 127 
 128 #define INCREMENTOR(name)           static void inc_##name() { _##name ++; }
 129 #define INCREMENTOR_ATOMIC(name)    static void inc_##name() { Atomic::inc(&_##name); }
 130 
 131   ALL_MY_COUNTERS(INCREMENTOR, INCREMENTOR_ATOMIC)
 132 
 133   static void print_on(outputStream* st);
 134 
 135 #undef INCREMENTOR
 136 #undef INCREMENTOR_ATOMIC
 137 
 138 };
 139 
 140 } // namespace metaspace
 141 
 142 #endif // ASSERT
 143 
 144 #endif // SHARE_MEMORY_METASPACE_INTERNSTAT_HPP
